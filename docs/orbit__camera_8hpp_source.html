61  void move(float x_dir, float y_dir, float z_dir) override {
62  float scale = glm::abs(target.z * map_size.x / 1e5);
63  target.x += x_dir * scale;
64  target.y += y_dir * scale;
65  target.y = glm::clamp(target.y, 0.f, map_size.y);
66  target.z += z_dir * scale;
67  target.z = glm::clamp(target.z, 0.f, map_size.x / 2.f);
68  }

70  void update(void) override {
71  map_position = chase.move_toward(map_position, target);
72  map_position.y = glm::clamp(map_position.y, 0.f, map_size.y);
73  map_position.z = glm::clamp(map_position.z, 0.f, map_size.x / 2.f);
74 
75  glm::vec3 normalized_pos = map_position / glm::vec3(map_size.x, map_size.y, map_size.x / 2.f);
76  glm::vec2 radiance_pos;
77  float pi = glm::pi<float>();
78  radiance_pos.x = glm::mod(normalized_pos.x * 2.f * pi, 2.f * pi);
79  radiance_pos.y = glm::max(0.f, glm::min(pi, normalized_pos.y * pi));
80 
81  float distance = radius + normalized_pos.z * radius * 7.;
82  world_position.x = distance * cos(radiance_pos.x) * sin(radiance_pos.y);
83  world_position.y = distance * sin(radiance_pos.x) * sin(radiance_pos.y);
84  world_position.z = distance * cos(radiance_pos.y);
85  };

100  glm::mat4 get_view() override {
101  glm::vec3 look_at = glm::vec3(0);
102  glm::vec3 up_vector = glm::vec3(0.f, -1.f, 0.f);
103 
104  glm::vec3 normalized_pos = map_position;

106  normalized_pos = normalized_pos / glm::vec3(map_size.x, map_size.y, radius);
107  glm::vec2 radiance_pos;
108  float pi = glm::pi<float>();
109  radiance_pos.x = glm::mod(normalized_pos.x * 2.f * pi, 2.f * pi);
110  radiance_pos.y = glm::max(0.f, glm::min(pi, normalized_pos.y * pi));
111  up_vector.x = -cos(radiance_pos.x) * cos(radiance_pos.y);
112  up_vector.y = -sin(radiance_pos.x) * cos(radiance_pos.y);
113  up_vector.z = sin(radiance_pos.y);

115  };

123  bool get_cursor_map_pos(std::pair<int, int> mouse_pos, glm::ivec2& out_pos) override {
124  float mouse_x = mouse_pos.first;
125  float mouse_y = screen_size.y - 1.f - mouse_pos.second;
126 
127  glm::mat4 view = get_view();
128  glm::mat4 projection = get_projection();
129 
130  glm::vec3 world_space_near = glm::unProject(
131  glm::vec3(mouse_x, mouse_y, 0.f),
132  view, projection,
133  glm::vec4(0.f, 0.f, screen_size));
134 
135  glm::vec3 world_space_far = glm::unProject(
136  glm::vec3(mouse_x, mouse_y, 1.f),
137  view, projection,
138  glm::vec4(0.f, 0.f, screen_size)
139  );
140 
141  glm::vec3 ray_direction = world_space_far - world_space_near;
142  ray_direction = glm::normalize(ray_direction);
143 
144  float distance = 0.f;
145  bool hit = glm::intersectRaySphere(world_space_near, ray_direction, glm::vec3(0, 0, 0), radius * radius, distance);
146 
147  glm::vec3 intersection_point = world_space_near + ray_direction * distance;
148  float pi = glm::pi<float>();
149  float y_rad = glm::acos(intersection_point.z / radius);
150  float x_rad = glm::atan(intersection_point.y, intersection_point.x);
151 
152  out_pos.x = map_size.x * x_rad / (2.f * pi);
153  out_pos.y = map_size.y * y_rad / (pi);
154 
155  return hit;
156  };
