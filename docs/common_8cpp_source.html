176 TechnologyInfo::TechnologyInfo(GameState& _gs, int x, int y, Technology* _technology, UI::Widget* parent)
177  : UI::Group(x, y, parent->width, 48, parent),
178  gs{ _gs },
179  technology{ _technology }
180 {
181  is_scroll = false;
182 
183  auto* chk = new UI::Checkbox(0, 0, 128, 24, this);
184  chk->text(technology->name);
185  chk->tooltip = new UI::Tooltip(chk, 512, 24);
186  chk->on_each_tick = ([](UI::Widget& w, void*) {
187  auto& o = static_cast<TechnologyInfo&>(*w.parent);
188  if(o.technology == o.gs.curr_nation->focus_tech || !o.gs.curr_nation->research[o.gs.world->get_id(o.technology)]) {
189  ((UI::Checkbox&)w).value = true;
190  } else {
191  ((UI::Checkbox&)w).value = false;
192  }
193 
194  if(o.gs.curr_nation->can_research(o.technology)) {
195  w.tooltip->text(UnifiedRender::Locale::translate("We can research this"));
196  } else {
197  std::string text = "";
198  text = UnifiedRender::Locale::translate("We can't research this because we don't have ");
199  for(const auto& req_tech : o.technology->req_technologies) {
200  if(o.gs.curr_nation->research[o.gs.world->get_id(req_tech)] > 0.f) {
201  text += UnifiedRender::Locale::translate(req_tech->name) + ", ";
202  }
203  }
204  w.tooltip->text(text);
205  }
206  });
207  chk->on_click = ([](UI::Widget& w, void*) {
208  auto& o = static_cast<TechnologyInfo&>(*w.parent);
209  if(o.gs.curr_nation->can_research(o.technology)) {
210  o.gs.client->send(Action::FocusTech::form_packet(o.technology));
211  }
212  });
213  chk->on_each_tick(*chk, nullptr);
214 
215  auto* pgbar = new UI::ProgressBar(0, 24, 128, 24, 0.f, technology->cost, this);
216  pgbar->on_each_tick = ([](UI::Widget& w, void*) {
217  auto& o = static_cast<TechnologyInfo&>(*w.parent);
218  ((UI::ProgressBar&)w).value = std::fabs(o.gs.curr_nation->research[o.gs.world->get_id(o.technology)] - o.technology->cost);
219  });
220 }
