47 // TODO: Make some form of "WarParticipationRequest" so we don't force allies to join
48 // and we also make sure betrayals are possible
49 void Nation::declare_war(Nation& nation, std::vector<TreatyClause::BaseClause*> clauses) {
50  World& world = World::get_instance();
51  auto* war = new War();
52 
53  UnifiedRender::Log::debug("game", ref_name + " has declared war on " + nation.ref_name);
54 
55  war->wargoals = clauses;
56 
57  // Recollect offenders
58  // - Those who are allied to us
59  for(unsigned int i = 0; i < this->relations.size(); i++) {
60  const auto& relation = this->relations[i];
61 
62  if(relation.has_alliance) {
63  war->attackers.push_back(world.nations[i]);
64  }
65  }
66  war->attackers.push_back(this);
67 
68  UnifiedRender::Log::debug("game", "Attackers");
69  for(const auto& attacker : war->attackers) {
70  UnifiedRender::Log::debug("game", attacker->ref_name);
71  }
72 
73  // Recollect defenders
74  // - Those who are on a defensive pact with the target
75  // - Those who are allied with the target
76  for(unsigned int i = 0; i < nation.relations.size(); i++) {
77  const auto& relation = nation.relations[i];
78 
79  if(relation.has_alliance || relation.has_defensive_pact) {
80  war->defenders.push_back(world.nations[i]);
81  }
82  }
83  war->defenders.push_back(&nation);
84 
85  // Attackers are at war with the defenders
86  for(auto& attacker : war->attackers) {
87  for(auto& defender : war->defenders) {
88  // Bilateral war
89  attacker->relations[world.get_id(defender)].has_war = true;
90  attacker->relations[world.get_id(defender)].has_alliance = false;
91  attacker->relations[world.get_id(defender)].has_defensive_pact = false;
92 
93  defender->relations[world.get_id(attacker)].has_war = true;
94  defender->relations[world.get_id(attacker)].has_alliance = false;
95  defender->relations[world.get_id(attacker)].has_defensive_pact = false;
96  }
97  }
98 
99  UnifiedRender::Log::debug("game", "Defenders");
100  for(const auto& defender : war->defenders) {
101  UnifiedRender::Log::debug("game", defender->ref_name);
102  }
103 
104  war->name = "War of " + this->name + " against " + nation.name;
105  UnifiedRender::Log::debug("game", war->name);
106 
107  world.insert(war);
108 }

177 // Enacts a policy on a nation
178 // @return false if policy draft failed to be applied, true if policy passed and is in-effect
179 void Nation::set_policy(Policies& policies) {
180  // No parliament? No referendum
181  if(current_policy.legislative_parliament != true) {
182  this->current_policy = policies;
183  UnifiedRender::Log::debug("game", "Parliament-less policy passed!");
184  return;
185  }
186 
187  unsigned int approvals = 0, disapprovals = 0;
188  std::vector<Pop*> disapprovers, approvers;
189  for(const auto& province : this->owned_provinces) {
190  for(auto& pop : province->pops) {
191  // Must have the minimum required social value
192  // the min-social-value is taken from the new enacted policy
193  if(pop.type->social_value < policies.min_sv_for_parliament) {
194  continue;
195  }
196 
197  const Policies& pop_policies = pop.get_ideology()->policies;
198 
199  // Disapproval of old (current) policy
200  const int old_disapproval = current_policy.difference(pop_policies);
201 
202  // Dissaproval of new policy
203  const int new_disapproval = policies.difference(pop_policies);
204 
205  if(new_disapproval < old_disapproval) {
206  approvals += pop.size;
207  disapprovers.push_back(&pop);
208  }
209  else {
210  disapprovals += pop.size;
211  approvers.push_back(&pop);
212  }
213  }
214  }
215 
216  // Policy is enacted and passed parliament
217  if(approvals > disapprovals) {
218  // Set new policy
219  this->current_policy = policies;
220 
221  // All people who agreed gets happy
222  for(auto& pop : approvers) {
223  pop->militancy /= std::min(pop->con, 0.1f);
224  }
225 
226  // All people who disagreed gets angered
227  for(auto& pop : disapprovers) {
228  pop->militancy *= std::min(pop->con, 0.1f);
229  }
230  UnifiedRender::Log::debug("game", "New enacted policy passed parliament!");
231  }
232  // Legislation does not make it into the official law
233  else {
234  // All people who agreed gets angered
235  for(auto& pop : approvers) {
236  pop->militancy *= std::min(pop->con, 0.1f);
237  }
238 
239  // All people who disagreed gets happy
240  for(auto& pop : disapprovers) {
241  pop->militancy /= std::min(pop->con, 0.1f);
242  }
243  UnifiedRender::Log::debug("game", "New enacted policy did not made it into the parliament!");
244  }
245  return;
246 }

266 // Gets the total tax applied to a POP depending on their "wealth"
267 // (not exactly like that, more like by their type/status)
268 float Nation::get_tax(const Pop& pop) const {
269  float base_tax = 1.f;
270 
271  if(!is_accepted_culture(pop) && !is_accepted_religion(pop)) {
272  // Exterminate imposes a scale of 3(+1), which will be enough to drive off most POPs
273  const int scale = 1 + current_policy.treatment;
274  base_tax *= 2 * scale;
275  }
276 
277  if(pop.type->social_value <= 1.f) {
278  return current_policy.poor_flat_tax * base_tax;
279  }
280  // For the medium class
281  else if(pop.type->social_value <= 2.f) {
282  return current_policy.med_flat_tax * base_tax;
283  }
284  // For the high class
285  else if(pop.type->social_value <= 3.f) {
286  return current_policy.rich_flat_tax * base_tax;
287  }
288  return base_tax;
289 }
