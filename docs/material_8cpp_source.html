59 std::vector<std::pair<UnifiedRender::Material*, std::string>> UnifiedRender::MaterialManager::load_wavefront(const std::string& path) {
60  std::ifstream file(path);
61  std::string line;
62 
63  print_info("Loading material file %s", path.c_str());
64 
65  std::vector<std::pair<UnifiedRender::Material*, std::string>> tmp_mat;
66  UnifiedRender::Material* curr_mat = nullptr;
67  while(std::getline(file, line)) {
68  // Skip whitespace
69  size_t len = line.find_first_not_of(" \t");
70  if(len != std::string::npos) {
71  line = line.substr(len, line.length() - len);
72  }
73 
74  // Comment
75  if(line[0] == '#' || line.empty()) {
76  continue;
77  }
78 
79  std::istringstream sline(line);
80  std::string cmd;
81  sline >> cmd;
82  if(cmd != "newmtl" && curr_mat == nullptr) {
83  continue;
84  }
85 
86  if(cmd == "newmtl") {
87  std::string name;
88  sline >> name;
89  curr_mat = new UnifiedRender::Material();
90  tmp_mat.push_back(std::make_pair(curr_mat, name));
91  } else if(cmd == "Ka") {
92  glm::vec3 color;
93  sline >> color.x >> color.y >> color.z;
94  curr_mat->ambient_color = color;
95  } else if(cmd == "Kd") {
96  glm::vec3 color;
97  sline >> color.x >> color.y >> color.z;
98  curr_mat->diffuse_color = color;
99  } else if(cmd == "Ks") {
100  glm::vec3 color;
101  sline >> color.x >> color.y >> color.z;
102  curr_mat->specular_color = color;
103  } else if(cmd == "Ns") {
104  sline >> curr_mat->specular_exp;
105  } else if(cmd == "Ni") {
106  sline >> curr_mat->optical_density;
107  } else if(cmd == "map_Kd") {
108  std::string map_path;
109  sline >> map_path;
110 
111  if(map_path[0] == '.') {
112  continue;
113  }
114  curr_mat->diffuse_map = &UnifiedRender::State::get_instance().tex_man->load(Path::get("3d/" + map_path));
115  } else {
116  print_info("Command %s not implemented", cmd.c_str());
117  }
118  }
119 
120  std::vector<std::pair<UnifiedRender::Material *, std::string>>::const_iterator mat;
121  for(mat = tmp_mat.begin(); mat != tmp_mat.end(); mat++) {
122  materials[(*mat).second] = (*mat).first;
123  }
124  return tmp_mat;
125 }
