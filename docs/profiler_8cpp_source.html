96 void BenchmarkTask::clear_old() {
97  // Clear old times
98  auto now = std::chrono::system_clock::now();
99  auto times_it = times.begin();
100  auto start_times_it = start_times.begin();
101  while(start_times_it != start_times.end()) {
102  auto prev_start_time = *start_times_it;
103  float time = std::chrono::duration_cast<std::chrono::milliseconds>(now - prev_start_time).count();
104  if(time > MS_TO_KEEP_DATA) {
105  start_times_it = start_times.erase(start_times_it);
106  times_it = times.erase(times_it);
107  } else {
108  start_times_it++;
109  times_it++;
110  }
111  }
112 }

145 void Profiler::render_done() {
146  auto now = std::chrono::system_clock::now();
147  if(!render_started) {
148  fps_clock = now;
149  render_started = true;
150  }
151  fps_timer += std::chrono::duration_cast<std::chrono::milliseconds>(now - fps_clock).count();
152  fps_clock = now;
153  frames++;
154  if(fps_timer > 1000.0) { //every second
155  fps = (float)frames * 0.5 + fps * 0.5; //more stable
156  frames = 0;
157  fps_timer -= 1000.;
158  }
159 }
