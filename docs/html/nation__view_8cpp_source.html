38 NationMarketView::NationMarketView(GameState& _gs, Nation* _nation)
39  : UI::Window(0, 0, 700, 600),

41  nation{ _nation }

43  //this->is_scroll = false;
44  this->text(UnifiedRender::Locale::translate("Market information"));
45 
46  unsigned int i = 0;

54  }
