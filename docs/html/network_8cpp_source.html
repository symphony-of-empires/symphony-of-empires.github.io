123 void UnifiedRender::Networking::SocketStream::send(const void* data, size_t size) {
124  const char* c_data = (const char*)data;
125  for(size_t i = 0; i < size; ) {
126  int r = ::send(fd, &c_data[i], std::min<size_t>(1024, size - i), 0);
127  if(r < 0)

150 int UnifiedRender::Networking::ServerClient::try_connect(int fd) {
151  sockaddr_in client;
152  socklen_t len = sizeof(client);
153  try {
154  conn_fd = accept(fd, (sockaddr*)&client, &len);
155  if(conn_fd == INVALID_SOCKET) {
156  throw UnifiedRender::Networking::SocketException("Cannot accept client connection");
157  }
158 
159  // At this point the client's connection was accepted - so we only have to check
160  // Then we check if the server is running and we throw accordingly
161  is_connected = true;
162 
163  print_info("New client connection established");
164  return conn_fd;
165  } catch(UnifiedRender::Networking::SocketException& e) {
166  // Continue
167  }
168  std::this_thread::sleep_for(std::chrono::seconds(3));
169  return 0;
170 }

172 // Push pending_packets to the packets queue (the packets queue is managed
173 // by us and requires almost 0 locking, so the host does not stagnate when
174 // trying to send packets to a certain client)
175 void UnifiedRender::Networking::ServerClient::flush_packets(void) {
176  if(!pending_packets.empty()) {
177  if(pending_packets_mutex.try_lock()) {
178  std::scoped_lock<std::mutex> lock(packets_mutex);
179 
180  std::deque<UnifiedRender::Networking::Packet>::iterator packet;
181  for(packet = pending_packets.begin(); packet != pending_packets.end(); packet++) {
182  packets.push_back(*packet);
183  }
184  pending_packets.clear();
185  pending_packets_mutex.unlock();
186  }
187  }
188 }

190 bool UnifiedRender::Networking::ServerClient::has_pending(void) {
191  // Check if we need to read packets
192 #ifdef unix
193  struct pollfd pfd = {};
194  pfd.fd = conn_fd;
195  pfd.events = POLLIN;
196 
197  int has_pending = poll(&pfd, 1, 10);
198  if(pfd.revents & POLLIN || has_pending) {
199  return true;
200  }
201 #elif defined windows
202  u_long has_pending = 0;
203  int test = ioctlsocket(conn_fd, FIONREAD, &has_pending);
204  if(has_pending) {
205  return true;
206  }
207 #endif
208  return false;
209 }

218 UnifiedRender::Networking::Server::Server(const unsigned port, const unsigned max_conn)
219  : n_clients{ static_cast<int>(max_conn) }
220 {
221 #ifdef windows
222  WSADATA data;
223  WSAStartup(MAKEWORD(2, 2), &data);
224 #endif
225 
226  std::memset(&addr, 0, sizeof(addr));
227  addr.sin_family = AF_INET;
228  addr.sin_addr.s_addr = htonl(INADDR_ANY);
229  addr.sin_port = htons(port);
230 
231  fd = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
232  if(fd == INVALID_SOCKET) {
233  throw UnifiedRender::Networking::SocketException("Cannot create server socket");
234  }
235 
236 #ifdef unix
237  int enable = 1;
238  setsockopt(fd, SOL_SOCKET, SO_REUSEADDR, &enable, sizeof(int));
239  setsockopt(fd, SOL_SOCKET, SO_REUSEPORT, &enable, sizeof(int));
240 #endif
241  if(bind(fd, (sockaddr*)&addr, sizeof(addr)) != 0) {
242  throw UnifiedRender::Networking::SocketException("Cannot bind server");
243  }
244 
245  if(listen(fd, max_conn) != 0) {
246  throw UnifiedRender::Networking::SocketException("Cannot listen in specified number of concurrent connections");
247  }
248 
249 #ifdef unix
250  // Allow non-blocking operations on this socket (we don't want to block on multi-listener servers)
251  fcntl(fd, F_SETFL, fcntl(fd, F_GETFD, 0) | O_NONBLOCK);
252 
253  // We need to ignore pipe signals since any client disconnecting **will** kill the server
254  signal(SIGPIPE, SIG_IGN);
255 #endif
256 
257  run = true;
258  print_info("Server created sucessfully and listening to %u; now invite people!", port);
259 }

272 // This will broadcast the given packet to all clients currently on the server
273 void UnifiedRender::Networking::Server::broadcast(const UnifiedRender::Networking::Packet& packet) {
274  for(size_t i = 0; i < n_clients; i++) {
275  if(clients[i].is_connected == true) {
276  // If we can "acquire" the spinlock to the main packet queue we will push
277  // our packet there, otherwise we take the alternative packet queue to minimize
278  // locking between server and client
279  if(clients[i].packets_mutex.try_lock()) {
280  clients[i].packets.push_back(packet);
281  clients[i].packets_mutex.unlock();
282  } else {
283  std::scoped_lock lock(clients[i].pending_packets_mutex);
284  clients[i].pending_packets.push_back(packet);
285  }
286 
287  // Disconnect the client when more than 200 MB is used
288  // we can't save your packets buddy - other clients need their stuff too!
289  size_t total_size = 0;
290 
291  std::deque<UnifiedRender::Networking::Packet>::const_iterator packet_q;
292  for(packet_q = clients[i].pending_packets.cbegin(); packet_q != clients[i].pending_packets.cend(); packet_q++) {
293  total_size += (*packet_q).buffer.size();
294  }
295 
296  if(total_size >= 200 * 1000000) {
297  clients[i].is_connected = false;
298  print_error("Client %zu has exceeded max quota! - It has used %zu bytes!", i, total_size);
299  }
300  }
301  }
302 }
