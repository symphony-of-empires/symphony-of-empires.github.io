69 // This dummy texture helps to avoid crashes due to missing buffers or so, and also gives
70 // visual aid of errors
71 void UnifiedRender::Texture::create_dummy() {
72  width = 8;
73  height = 8;
74  buffer = std::unique_ptr<uint32_t>(new uint32_t[width * height]);
75  if(buffer == nullptr) {
76  throw TextureException("Dummy", "Out of memory for dummy texture");
77  }
78 
79  // Fill in with a permutation pattern of pink and black
80  // This should be autovectorized by gcc
81  for(size_t i = 0; i < width * height; i++) {
82  buffer.get()[i] = 0xff000000 | (i * 16);
83  }
84 }

86 void UnifiedRender::Texture::to_opengl(TextureOptions options) {
87  if(gl_tex_num) {
88  delete_opengl();
89  }
90 
91  glGenTextures(1, &gl_tex_num);
92  glBindTexture(GL_TEXTURE_2D, gl_tex_num);
93  glTexImage2D(GL_TEXTURE_2D, 0, options.internal_format, width, height, 0, options.format, options.type, buffer.get());
94 
95  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, options.wrap_s);
96  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, options.wrap_t);
97  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, options.min_filter);
98  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, options.mag_filter);
99 
100  // We will free up the texture if we don't plan on editing it since it's on the GPU now
101  if(!options.editable) {
102  buffer.reset();
103  }
104 }

111 // Converts the texture into a OpenGL texture, and assigns it a number
112 void UnifiedRender::Texture::to_opengl(SDL_Surface* surface) {
113  int colors = surface->format->BytesPerPixel;
114  GLuint texture_format;
115  if(colors == 4) {
116  // Alpha
117  if(surface->format->Rmask == 0x000000ff) {
118  texture_format = GL_RGBA;
119  } else {
120  texture_format = GL_BGRA;
121  }
122  } else {
123  // No alpha
124  if(surface->format->Rmask == 0x000000ff) {
125  texture_format = GL_RGB;
126  } else {
127  texture_format = GL_BGR;
128  }
129  }
130 
131  // On some systems OpenGL only allows powers of 2
132 
133  glGenTextures(1, &gl_tex_num);
134  glBindTexture(GL_TEXTURE_2D, gl_tex_num);
135  glTexImage2D(GL_TEXTURE_2D, 0, colors, surface->w, surface->h, 0, texture_format, GL_UNSIGNED_BYTE, surface->pixels);
136 
137  if(glewIsSupported("GL_VERSION_2_1")) {
138  glGenerateMipmap(GL_TEXTURE_2D);
139  } else {
140  glTexParameteri(GL_TEXTURE_2D, GL_GENERATE_MIPMAP, GL_TRUE);
141  }
142 
143  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
144  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
145  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
146  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
147 }

173 // Uploads the TextureArray to the driver
174 void UnifiedRender::TextureArray::to_opengl(GLuint wrapp, GLuint min_filter, GLuint mag_filter) {
175  glGenTextures(1, &gl_tex_num);
176  glBindTexture(GL_TEXTURE_2D_ARRAY, gl_tex_num);
177 
178  // set up texture handle parameters
179  // glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_BASE_LEVEL, 0); // !single image!
180  // glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_MAX_LEVEL, 1); // !single image! mat->mips == 1
181  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
182  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
183  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_WRAP_S, GL_REPEAT);
184  glTexParameteri(GL_TEXTURE_2D_ARRAY, GL_TEXTURE_WRAP_T, GL_REPEAT);
185 
186  size_t p_dx = width / tiles_x; // pixels of each tile in x
187  size_t p_dy = height / tiles_y; // pixels of each tile in y
188  glTexImage3D(GL_TEXTURE_2D_ARRAY, 0, GL_RGBA, p_dx, p_dy, 16, 0, GL_RGBA, GL_UNSIGNED_BYTE, NULL);
189 
190  glPixelStorei(GL_UNPACK_ROW_LENGTH, width);
191  glPixelStorei(GL_UNPACK_IMAGE_HEIGHT, height);
192 
193  for(size_t x = 0; x < tiles_x; x++) {
194  for(size_t y = 0; y < tiles_y; y++) {
195  glTexSubImage3D(GL_TEXTURE_2D_ARRAY, 0, 0, 0, x * tiles_x + y, p_dx, p_dy, 1, GL_RGBA, GL_UNSIGNED_BYTE, buffer.get() + (x * p_dy * width + y * p_dx));
196  }
197  }
198  //glGenerateMipmap(GL_TEXTURE_2D_ARRAY);
199 
200  // unbind texture handle
201  glBindTexture(GL_TEXTURE_2D_ARRAY, 0);
202  glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);
203  glPixelStorei(GL_UNPACK_IMAGE_HEIGHT, 0);
204 }

210 //
211 // Finds a texture in the list of a texture manager
212 // if the texture is already in the list we load the saved texture from the list
213 // instead of loading it from the disk.
214 //
215 // Otherwise we load it from the disk and add it to the saved texture list
216 //
217 // The object returned is a pointer and we will not give ownership of textures in the list
218 // and program should not modify the contents of it since it will differ from the texture
219 // on the disk, and our main point is to mirror loaded textures from the disk - not modify
220 // them.
221 //
222 const UnifiedRender::Texture& UnifiedRender::TextureManager::load(const std::string& path, TextureOptions options) {
223  // Find texture when wanting to be loaded and load texture from cached texture list
224  std::map<std::string, UnifiedRender::Texture*>::const_iterator it = textures.find(path);
225  if(it != textures.end()) {
226  return *((*it).second);
227  }
228 
229  print_info("Loaded and cached texture %s", path.c_str());
230 
231  // Otherwise texture is not in our control, so we create a new texture
232  UnifiedRender::Texture* tex;
233  try {
234  tex = new UnifiedRender::Texture(path);
235  } catch(BinaryImageException&) {
236  tex = new UnifiedRender::Texture();
237  tex->create_dummy();
238  }
239 
240  tex->to_opengl(options);
241  if(options.min_filter == GL_NEAREST_MIPMAP_NEAREST || options.min_filter == GL_NEAREST_MIPMAP_LINEAR || options.min_filter == GL_LINEAR_MIPMAP_NEAREST || options.min_filter == GL_LINEAR_MIPMAP_LINEAR) {
242  tex->gen_mipmaps();
243  }
244  textures[path] = tex;
245  return *((const Texture*)tex);
246 }
