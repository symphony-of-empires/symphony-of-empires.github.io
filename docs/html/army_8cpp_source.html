111 ArmyProductionTab::ArmyProductionTab(GameState& _gs, int x, int y, UI::Widget* parent)
112  : UI::Group(x, y, parent->width - x, parent->height, parent),
113  gs{ _gs }
114 {
115  // Chart showing total number of required materials
116  this->reqmat_chart = new UI::Chart(0, 0, 128, 128, this);
117  this->reqmat_chart->text("Material demand");
118  this->reqmat_chart->on_each_tick = ([](UI::Widget& w, void*) {
119  auto& o = static_cast<ArmyProductionTab&>(*w.parent);
120 
121  float reqtotal = 0.f;
122  for(const auto& building : o.gs.world->buildings) {
123  if(building->get_owner() != o.gs.curr_nation) {
124  continue;
125  }
126 
127  for(const auto& req : building->req_goods_for_unit) {
128  reqtotal += req.second;
129  }
130 
131  for(const auto& req : building->req_goods) {
132  reqtotal += req.second;
133  }
134  }
135 
136  o.reqmat_chart->data.push_back(reqtotal);
137  if(o.reqmat_chart->data.size() >= 30) {
138  o.reqmat_chart->data.pop_back();
139  }
140  });
141 
142  unsigned int i = 0;
143  for(const auto& building : gs.world->buildings) {
144  if(building->get_owner() != gs.curr_nation) {
145  continue;
146  }
147 
148  new ArmyProductionUnitInfo(gs, 0, 128 + (48 * i), building, this);
149  i++;
150  }
151 }

154  : UI::Group(x, y, parent->width - x, 48, parent),

176  this->name_lab->right_side_of(*this->province_lab);
177  this->name_lab->on_each_tick = ([](UI::Widget& w, void*) {
178  auto& o = static_cast<ArmyProductionUnitInfo&>(*w.parent);
179  w.text((o.building->working_unit_type != nullptr) ? UnifiedRender::Locale::translate(o.building->working_unit_type->name) : "No unit");
180  });
181  this->name_lab->on_each_tick(*this->name_lab, nullptr);
182 
183  auto* progress_pgbar = new UI::ProgressBar(0, 0, 128, 24, 0.f, 1.f, this);
184  progress_pgbar->below_of(*this->name_lab);
185  progress_pgbar->on_each_tick = ([](UI::Widget& w, void*) {
186  auto& o = static_cast<ArmyProductionUnitInfo&>(*w.parent);
187 
188  std::string text = "";
189 
190  size_t full = 0, needed = 0;
191  text = "Needs ";
192  for(size_t i = 0; i < o.building->req_goods_for_unit.size(); i++) {
193  auto need_req = o.building->req_goods_for_unit[i];
194  auto full_req = o.building->working_unit_type->req_goods[i];
195 
196  full_req.second -= need_req.second;
197 
198  full += full_req.second;
199  needed += need_req.second;
200 
201  text += std::to_string(need_req.second) + " of " + UnifiedRender::Locale::translate(need_req.first->name) + " (has " + std::to_string(full_req.second) + "), ";
202  }
203 
204  ((UI::ProgressBar&)w).value = (float)full / (float)needed;
205  w.tooltip->text(text);
206  });
207  progress_pgbar->tooltip = new UI::Tooltip(progress_pgbar, 512, 24);
208  progress_pgbar->on_each_tick(*progress_pgbar, nullptr);
209 }
