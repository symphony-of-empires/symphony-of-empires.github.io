179 glm::ivec2 Context::get_pos(Widget& w, glm::ivec2 offset) {
180  glm::ivec2 pos{ w.x, w.y };
181  glm::ivec2 screen_size{ width, height };
182  glm::ivec2 parent_size{ 0, 0 };
183  if(w.parent != nullptr) {
184  parent_size = glm::ivec2{ w.parent->width, w.parent->height };
185  }
186 
187  switch(w.origin) {
188  case UI::Origin::CENTER:
189  pos += offset;
190  pos += parent_size / 2;
191  break;
192  case UI::Origin::UPPER_LEFT:
193  pos += offset;
194  break;
195  case UI::Origin::UPPER_MIDDLE:
196  pos += offset;
197  pos.x += parent_size.x / 2;
198  break;
199  case UI::Origin::UPPER_RIGTH:
200  pos += offset;
201  pos.x += parent_size.x;
202  break;
203  case UI::Origin::LOWER_LEFT:
204  pos += offset;
205  pos.y += parent_size.y;
206  break;
207  case UI::Origin::LOWER_MIDDLE:
208  pos += offset;
209  pos.y += parent_size.y;
210  pos.x += parent_size.x / 2;
211  break;
212  case UI::Origin::LOWER_RIGHT:
213  pos += offset;
214  pos += parent_size;
215  break;
216  case UI::Origin::CENTER_SCREEN:
217  pos += screen_size / 2;
218  break;
219  case UI::Origin::UPPER_LEFT_SCREEN:
220  break;
221  case UI::Origin::UPPER_MIDDLE_SCREEN:
222  pos.x += screen_size.x / 2;
223  break;
224  case UI::Origin::UPPER_RIGHT_SCREEN:
225  pos.x += screen_size.x;
226  break;
227  case UI::Origin::LOWER_LEFT_SCREEN:
228  pos.y += screen_size.y;
229  break;
230  case UI::Origin::LOWER_MIDDLE_SCREEN:
231  pos.y += screen_size.y;
232  pos.x += screen_size.x / 2;
233  break;
234  case UI::Origin::LOWER_RIGHT_SCREEN:
235  pos += screen_size;
236  break;
237  }
238  return pos;
239 }

246 void Context::render_recursive(Widget& w, UnifiedRender::Rect viewport) {
247  if(!w.is_show || !w.is_render) {
248  return;
249  }
250  if(!w.width || !w.height) {
251  return;
252  }
253 
254  if(w.is_fullscreen) {
255  w.width = width;
256  w.height = height;
257  }
258 
259  glm::ivec2 offset{ viewport.left, viewport.top };
260  glm::ivec2 size{ w.width, w.height };
261  offset = get_pos(w, offset);
262  UnifiedRender::Rect local_viewport = UnifiedRender::Rect{ offset, size };
263 
264  if (!w.parent || w.parent->type != UI::WidgetType::GROUP) {
265  local_viewport = viewport.intersection(local_viewport);
266  }
267  viewport = local_viewport;
268 
269  local_viewport.offset(-offset);
270  glPushMatrix();
271  glTranslatef(offset.x, offset.y, 0.f);
272  w.on_render(*this, local_viewport);
273  glPopMatrix();
274 
275  if(w.on_update) {
276  w.on_update(w, w.user_data);
277  }
278 
279  for(auto& child : w.children) {
280  child->is_show = true;
281  child->is_clickable = (w.on_click || w.is_clickable) && w.is_hover;
282  if(viewport.size().x <= 0 || viewport.size().y <= 0) {
283  if(!child->is_float) {
284  child->is_show = false;
285  }
286  }
287 
288  render_recursive(*child, viewport);
289  }
290 }

316 bool Context::check_hover_recursive(Widget& w, const unsigned int mx, const unsigned int my, int x_off, int y_off) {
317  glm::ivec2 offset{ x_off, y_off };
318  offset = get_pos(w, offset);
319 
320  w.is_hover = true;
321  if(!w.is_show || !w.is_render) {
322  return false;
323  }
324 
325  if(!((int)mx >= offset.x && mx <= offset.x + w.width && (int)my >= offset.y && my <= offset.y + w.height)) {
326  w.is_hover = false;
327  }
328  else if(w.is_transparent) {
329  if(w.current_texture != nullptr) {
330  int tex_width = w.current_texture->width;
331  int tex_height = w.current_texture->height;
332  int tex_x = ((mx - offset.x) * tex_width) / w.width;
333  int tex_y = ((my - offset.y) * tex_height) / w.height;
334  if(tex_x >= 0 && tex_x < tex_width && tex_y >= 0 && tex_y < tex_height) {
335  uint32_t argb = w.current_texture->get_pixel(tex_x, tex_y);
336  if(((argb >> 24) & 0xff) == 0) {
337  w.is_hover = false;
338  }
339  }
340  }
341  }
342 
343  bool consumed_hover = w.is_hover && w.type != UI::WidgetType::GROUP;
344  if(w.is_hover) {
345  if(w.on_hover) {
346  glm::ivec2 mouse_pos(mx, my);
347  w.on_hover(w, mouse_pos, offset);
348  }
349 
350  if(w.tooltip != nullptr) {
351  tooltip_widget = w.tooltip;
352  tooltip_widget->set_pos(offset.x, offset.y, w.width, w.height, width, height);
353  }
354 
355  for(auto& child : w.children) {
356  consumed_hover |= check_hover_recursive(*child, mx, my, offset.x, offset.y);
357  }
358  }
359  return consumed_hover;
360 }

365  std::pair<int, int> diff = std::make_pair(offset.first - dragged_widget->x, offset.second - dragged_widget->y);
366 
367  dragged_widget->move_by(diff.first, diff.second);

379 UI::ClickState Context::check_click_recursive(Widget& w, const unsigned int mx, const unsigned int my, int x_off, int y_off, UI::ClickState click_state, bool clickable) {
380  glm::ivec2 offset{ x_off, y_off };
381  offset = get_pos(w, offset);
382 
383  if(click_state != UI::ClickState::NOT_CLICKED) {
384  clickable = true;
385  }
386 
387  bool click_consumed = click_state == UI::ClickState::HANDLED;
388 
389  // Widget must be displayed
390  if(!w.is_show || !w.is_render) {
391  clickable = false;
392  }
393 
394  // Click must be within the widget's box if it's not a group
395  if(w.type != UI::WidgetType::GROUP) {
396  if(!((int)mx >= offset.x && mx <= offset.x + w.width && (int)my >= offset.y && my <= offset.y + w.height)) {
397  clickable = false;
398  }
399  else if(w.is_transparent) {
400  if(w.current_texture != nullptr) {
401  int tex_width = w.current_texture->width;
402  int tex_height = w.current_texture->height;
403  int tex_x = ((mx - offset.x) * tex_width) / w.width;
404  int tex_y = ((my - offset.y) * tex_height) / w.height;
405  if(tex_x >= 0 && tex_x < tex_width && tex_y >= 0 && tex_y < tex_height) {
406  uint32_t argb = w.current_texture->get_pixel(tex_x, tex_y);
407  if(((argb >> 24) & 0xff) == 0) {
408  clickable = false;
409  }
410  }
411  }
412  }
413  }
414 
415  for(auto& child : w.children) {
416  click_state = check_click_recursive(*child, mx, my, offset.x, offset.y, click_state, clickable);
417  }
418 
419  // Non-clickable group widgets are only taken in account
420  if(w.type == UI::WidgetType::GROUP && w.on_click == nullptr) {
421  clickable = false;
422  }
423 
424  // Call on_click_outside if on_click has been used or widget isn't hit by click
425  if(w.on_click_outside) {
426  if(!clickable || click_consumed) {
427  w.on_click_outside(w, w.user_data);
428  }
429  }
430 
431  // Call on_click if on_click hasnt been used and widget is hit by click
432  if(w.on_click) {
433  if(clickable && !click_consumed) {
434  if(w.type == UI::WidgetType::SLIDER) {
435  Slider* wc = (Slider*)&w;
436  wc->value = ((float)std::abs((int)mx - offset.x) / (float)wc->width) * wc->max;
437  }
438 
439  w.on_click(w, w.user_data);
440  return UI::ClickState::HANDLED;
441  }
442  }
443 
444  if(click_state == UI::ClickState::NOT_CLICKED && clickable) {
445  click_state = UI::ClickState::NOT_HANDLED;
446  }
447  return click_state;
448 }

450 bool Context::check_click(const unsigned mx, const unsigned my) {
451  is_drag = false;
452  UI::ClickState click_state = UI::ClickState::NOT_CLICKED;
453  int click_wind_index = -1;
454 
455  bool is_click = false;
456  for(int i = widgets.size() - 1; i >= 0; i--) {
457  click_state = check_click_recursive(*widgets[i], mx, my, 0, 0, click_state, true);
458 
459  if(click_state != UI::ClickState::NOT_CLICKED) {
460  is_click = true;
461  }
462 
463  // Check if windows should move to the top
464  if(click_wind_index == -1 && click_state != UI::ClickState::NOT_CLICKED) {
465  click_wind_index = i;
466  }
467  }
468 
469  if(click_wind_index != -1) {
470  // Only movable and UI::WidgetType::WINDOWS are able to move to the top
471  Widget* window = widgets[click_wind_index];
472  if(window->type == UI::WidgetType::WINDOW && !window->is_pinned) {
473  auto it = widgets.begin() + click_wind_index;
474  std::rotate(it, it + 1, widgets.end());
475  }
476  }
477  return is_click;
478 }

480 void Context::check_drag(const unsigned mx, const unsigned my) {
481  for(int i = widgets.size() - 1; i >= 0; i--) {
482  Widget& widget = *widgets[i];
483 
484  // Only windows can be dragged around
485  if(widget.type != UI::WidgetType::WINDOW) {
486  continue;
487  }
488 
489  // Pinned widgets are not movable
490  if(widget.is_pinned) {
491  continue;
492  }
493 
494  if((int)mx >= widget.x && mx <= widget.x + widget.width && (int)my >= widget.y && (int)my <= widget.y + 24) {
495  auto& c_widget = static_cast<Window&>(widget);
496  if(!c_widget.is_movable) {
497  continue;
498  }
499 
500  if(!is_drag) {
501  drag_x = mx - widget.x;
502  drag_y = my - widget.y;
503  is_drag = true;
504  dragged_widget = &widget;
505  break;
506  }
507  }
508  }
