89 static inline float euclidean_distance(const World& world, Tile* t1, Tile* t2) {
90  const size_t t1_idx = world.get_id(t1);

107 static inline float tile_cost(const World& world, Tile* t1, Tile* t2) {
108  const size_t t1_idx = world.get_id(t1);
109  const size_t t1_x = t1_idx % world.width;
110  const size_t t1_y = t1_idx / world.width;
111 
112  const size_t t2_idx = world.get_id(t2);
113  const size_t t2_x = t2_idx % world.width;
114  const size_t t2_y = t2_idx / world.width;
115 
116  const int x_diff = t1_x - t2_x;
117  const int y_diff = t1_y - t2_y;
118 
119  // TODO: Terrain penalties apply PER UNIT TYPE
120  // a boat can pass thru water but for an army it will have a horrible cost!
121 
122  // Maximum penalties difference accounts to same cost as one jump in x or y direction (1.0)
123  const float elev_diff = (
124  (int)world.terrain_types[t1->terrain_type_id]->movement_penalty
125  - (int)world.terrain_types[t2->terrain_type_id]->movement_penalty)
126  / 128.f;
127 
128  // Base distance is euclidean distance in x, y and elevation
129  const float distance = std::sqrt(x_diff* x_diff + y_diff* y_diff + elev_diff* elev_diff);
130 
131  // Calculate average infrastructure level between the two tiles
132  return distance;

139 std::vector<Tile *> Pathfind::unit_path(const World& world, const Nation& nation, Tile* start, Tile* end) {
140  // We can't go to sea
141  //if(start->elevation <= world.sea_level && end->elevation <= world.sea_level) {
142  // return std::vector<Tile *>();
143  //}
144 
145  // Keeps track of the costs so far
146  std::unordered_map<Tile *, float> cost_map;
147 
148  // Keeps track of the previous tile for each tile
149  std::unordered_map<Tile *, Tile *> prev_map;
150 
151  // Keeps track of which tiles have been visited
152  std::unordered_set<Tile *> visited;
153 
154  // Priority queue based on cost
155  std::priority_queue<
156  std::pair<float, Tile *>,
157  std::vector<std::pair<float, Tile *>>,
158  std::greater<std::pair<float, Tile *>>
159  > queue;
160 
161  cost_map[start] = 0.f;
162  queue.push({0.f, start});
163 
164  while(!queue.empty()) {
165  Tile* current = queue.top().second;
166  queue.pop();
167 
168  // If the current node has been previously visited,
169  // it's optimal cost has already been calculated, and we can skip it
170  if(visited.count(current))
171  continue;
172 
173  visited.insert(current);
174 
175  // We are done
176  if(current == end)
177  break;
178 
179  // Generate neighbours
180  for(const auto neighbor: generate_neighbors(world, nation, current)) {
181  // If the neighbor is visited, we already have the optimal path to it
182  if(visited.count(neighbor))
183  continue;
184 
185  const float cost = cost_map[current] + tile_cost(world, current, neighbor);
186 
187  // If we found a new tile or a shorter path to a previously found tile
188  if(!cost_map.count(neighbor) || cost < cost_map[neighbor]) {
189  cost_map[neighbor] = cost;
190  const float priority = cost + euclidean_distance(world, neighbor, end);
191  queue.push({priority, neighbor});
192  prev_map[neighbor] = current;
193  }
194  }
195  }
196 
197  std::vector<Tile *> path;
198 
199  // Unwind path and reverse
200  Tile* current = end;
201  while(current != start) {
202  path.push_back(current);
203  current = prev_map[current];
204  }
205  path.push_back(start);
206 
207  std::reverse(path.begin(), path.end());
208  return path;
209 }
