46 // Obtain best potential good
47 Good* ai_get_potential_good(Nation* nation, World* world) {
48  // We will randomly determine if we want to do a market analysis for secondary and tertiary products or if
49  // we want to build primary-sector products (to kickstart chains)
50  if(!(std::rand() % 5)) {
51  // Analyze the market probability of sucess of a product this is determined by (demand * price), we calculate
52  // this from the average of all products on our provinces we currently own
53 
54  // Obtain the average probability of profitability per good we will only however, take in account products
55  // which are produced on provinces we own, hereafter we can obtain the average of the national good industry
56  print_info("Market analysis strategy");
57 
58  // So our formula would be:
59  // Sucess = Sum(Demand / (Supply + 1) * Price)
60  std::vector<float> avg_prob = std::vector<float>(world->goods.size(), 0.f);
61  for(const auto& product : world->products) {
62  if(product->building == nullptr || product->building->get_owner() != nation) {
63  continue;
64  }
65 
66  avg_prob[world->get_id(product->good)] += product->demand / (product->supply + 1) * product->price;
67  }
68 
69  // Now, we will take in account 2 factors:
70  // - First, an industry which takes-in a good with high sucess and outputs something should have that something
71  // have a sucess probability equal or higher to the taked good
72  // - Second, we preferably want the A.I to complete the supply chain of all industries so doing the method described
73  // above should increase the priority for filling out higher level industries
74  for(const auto& building_type : world->building_types) {
75  // Take in account all buildings for this
76  for(const auto& input : building_type->inputs) {
77  // Apply the higher-probability with outputs of this factory
78  for(const auto& output : building_type->outputs) {
79  avg_prob[world->get_id(output)] += avg_prob[world->get_id(input)] + 1;
80  }
81  }
82  }
83 
84  Good* target_good = world->goods.at(std::distance(avg_prob.begin(), std::max_element(avg_prob.begin(), avg_prob.end())));
85 
86  // The more buildings there are in the world the less we are wiling to construct one
87  float saturation = std::max<size_t>(1, world->buildings.size()) / 100;
88  if(fmod(std::rand(), saturation)) {
89  print_info("Too much market saturation");
90  return nullptr;
91  }
92 
93  // Obtain the index of the highest element (the one with more sucess rate)
94  return target_good;
95  }else {
96  // We will randomly pick any primary product which we are capable of producing
97  // This is mostly useful for starting supply chains from zero
98  print_info("Primary sector kickstart strategy");
99 
100  // The more buildings there are in the world the less we are wiling to construct one
101  // (more intense with primary sector due to primary-industry spam)
102  float saturation = std::max<size_t>(1, world->buildings.size()) / 50;
103  if(fmod(std::rand(), saturation)) {
104  print_info("Too much market saturation");
105  return nullptr;
106  }
107 
108  for(const auto& building_type : world->building_types) {
109  // Only take in account RGOs (and buildings that have atleast 1 output)
110  if(!building_type->inputs.empty() || !building_type->outputs.size()) {
111  continue;
112  }
113 
114  if(!building_type->is_factory) {
115  continue;
116  }
117 
118  // Randomness
119  if(std::rand() % 5) {
120  continue;
121  }
122  return building_type->outputs[std::rand() % building_type->outputs.size()];
123  }
124  }
125 
126  print_info("No suitable good");
127  return nullptr;
128 }

181 // Update relations with another nation
182 void ai_update_relations(Nation* nation, Nation* other) {
183  const World& world = World::get_instance();
184 
185  NationRelation& relation = nation->relations[world.get_id(other)];
186  NationRelation& other_relation = other->relations[world.get_id(nation)];
187 
188  // Try to increase relations with our friend
189  if(nation->is_ally(*other) && !(std::rand() % 250)) {
190  nation->increase_relation(*other);
191 
192  // Propose an alliance
193  if(relation.relation > 1.f && !other_relation.has_alliance) {
194  relation.has_alliance = true;
195  print_info("[%s] requested an alliance with [%s]!", nation->ref_name.c_str(), other->ref_name.c_str());
196  }
197 
198  // If they want an alliance we won't hesitate to join (they are our friends after all)
199  if(other_relation.has_alliance && !relation.has_alliance) {
200  relation.has_alliance = true;
201  print_info("Alliance [%s] <-> [%s] has been made!", nation->ref_name.c_str(), other->ref_name.c_str());
202  }
203 
204  // Same with the defensive pact
205  if(other_relation.has_defensive_pact && !relation.has_defensive_pact) {
206  relation.has_defensive_pact = true;
207  print_info("Defensive pact [%s] <-> [%s] has been made!", nation->ref_name.c_str(), other->ref_name.c_str());
208  }
209  }
210 
211  // Hate our enemies more
212  if(nation->is_enemy(*other) && !(std::rand() % 250)) {
213  nation->decrease_relation(*other);
214 
215  // Embargo them
216  if(relation.relation < -1.f) {
217  relation.has_embargo = true;
218  print_info("[%s] has placed an embargo on [%s]!", nation->ref_name.c_str(), other->ref_name.c_str());
219  }
220 
221  // We really hate our enemies, don't we?
222  if(relation.relation < -50.f && !relation.has_war) {
223  // TODO: Do not war if it's beyond our capabilities (i.e Liechestein vs. France, Prussia and UK)
224  nation->declare_war(*other);
225  }
226  }
227 
228  // Randomness to spice stuff up
229  if(!(std::rand() % 10)) {
230  nation->increase_relation(*other);
231  } else if(!(std::rand() % 10)) {
232  nation->decrease_relation(*other);
233  }
234 
235  // Check if we even border them
236  bool is_border = false;
237  for(const auto& province : nation->owned_provinces) {
238  for(const auto& neighbour : province->neighbours) {
239  if(neighbour->controller == other) {
240  is_border = true;
241  break;
242  }
243  }
244  }
245 
246  // Our strength as attackers
247  float our_power = 1.f;
248  for(const auto& ally_nation : nation->get_allies()) {
249  for(const auto& province : ally_nation->owned_provinces) {
250  for(const auto& unit : province->get_units()) {
251  our_power += unit->type->attack * unit->size;
252  our_power += unit->type->defense * unit->size;
253  }
254  }
255  }
256 
257  // The strength of the defenders
258  float other_power = 1.f;
259  for(const auto& ally_nation : other->get_allies()) {
260  for(const auto& province : ally_nation->owned_provinces) {
261  for(const auto& unit : province->get_units()) {
262  other_power += unit->type->attack * unit->size;
263  other_power += unit->type->defense * unit->size;
264  }
265  }
266  }
267 
268  // Hating a nation a lot will make us reconsider logic military actions and go "purely by instinct"
269  // Calculate the times the other nation has our power, multiply that by a factor of 1,000,000
270  // If the relation is negative then we divide by the positive sum of it
271  if(relation.relation < 10.f) {
272  const float force_dist = 10.f * ((1.f + other_power) / (1.f + our_power));
273  const int chance = std::max<int>(0, force_dist - -relation.relation);
274  if(std::rand() % (100 + (chance * 100)) == 0) {
275  if(!relation.has_war) {
276  nation->declare_war(*other);
277  }
278  }
279  }
280 
281  if(relation.has_war) {
282  // Offer treaties
283  if(!(std::rand() % 50)) {
284  Treaty* treaty = new Treaty();
285 
286  {
287  auto* clause = new TreatyClause::AnexxProvince();
288  clause->sender = nation;
289  clause->receiver = other;
290  for(const auto& province : other->owned_provinces) {
291  if(province->controller == nation) {
292  if(!(std::rand() % 10)) {
293  clause->provinces.push_back(province);
294  }
295  }
296  }
297 
298  if(!clause->provinces.empty()) {
299  treaty->clauses.push_back(clause);
300  }
301  }
302 
303  {
304  auto* clause = new TreatyClause::LiberateNation();
305  clause->sender = nation;
306  clause->receiver = other;
307 
308  clause->liberated = nullptr;
309  for(const auto& province : other->owned_provinces) {
310  for(const auto& other_nation : province->nuclei) {
311  if(other_nation != other) {
312  clause->liberated = other_nation;
313  break;
314  }
315  }
316 
317  if(clause->liberated != nullptr) {
318  break;
319  }
320  }
321 
322  if(clause->liberated != nullptr) {
323  treaty->clauses.push_back(clause);
324  }
325  }
326 
327  {
328  auto* clause = new TreatyClause::Ceasefire();
329  clause->sender = nation;
330  clause->receiver = other;
331  treaty->clauses.push_back(clause);
332  }
333 
334  ((World&)world).insert(treaty);
335  }
336  }
337 }

339 void ai_build_commercial(Nation* nation, World* world) {
340  Good* target_good;
341  target_good = ai_get_potential_good(nation, world);
342  if(target_good == nullptr) {
343  return;
344  }
345 
346  // Find an industry type which outputs this good
347  BuildingType* type = nullptr;
348  for(const auto& building_type : world->building_types) {
349  if(!building_type->is_factory) {
350  continue;
351  }
352 
353  /*for(const auto& input : building_type->inputs) {
354  if(input == target_good) {
355  type = (BuildingType*)building_type;
356  break;
357  }
358  }*/
359 
360  for(const auto& output : building_type->outputs) {
361  if(output == target_good) {
362  type = (BuildingType*)building_type;
363  break;
364  }
365  }
366  }
367 
368  print_info("[%s]: Good [%s] seems to be on a high-trend - building industry [%s] which makes that good", nation->ref_name.c_str(), target_good->ref_name.c_str(), type->ref_name.c_str());
369 
370  // Otherwise -- do not build anything since the highest valued good cannot be produced
371  if(type == nullptr) {
372  return;
373  }
374 
375  auto it = std::begin(nation->owned_provinces);
376  std::advance(it, std::rand() % nation->owned_provinces.size());
377 
378  Province* province = *it;
379  if(province->min_x > world->width || province->min_y == world->height || province->max_x < province->min_x || province->max_y < province->min_y || !province->n_tiles) {
380  print_error("Cant build buidling, province doesn't have any tiles");
381  } else {
382  // Now build the building
383  Building* building = new Building();
384  building->province = province;
385  building->type = world->building_types[0];
386  building->owner = nation;
387  building->budget = 100.f;
388  if(building->type->is_factory) {
389  building->create_factory();
390  for(const auto& product : building->output_products) {
391  UnifiedRender::Networking::Packet packet = UnifiedRender::Networking::Packet();
392  Archive ar = Archive();
393  ActionType action = ActionType::PRODUCT_ADD;
394  ::serialize(ar, &action);
395  ::serialize(ar, product);
396  packet.data(ar.get_buffer(), ar.size());
397  g_server->broadcast(packet);
398  }
399  }
400  world->insert(building);
401 
402  // Broadcast the addition of the building to the clients
403  g_server->broadcast(Action::BuildingAdd::form_packet(building));
404  print_info("Building of [%s](%i), from [%s] built on [%s]", building->type->ref_name.c_str(), (int)world->get_id(building->type), nation->ref_name.c_str(), building->get_province()->ref_name.c_str());
405  }

408 void ai_do_tick(Nation* nation, World* world) {
409  if(!nation->exists() || !nation->owned_provinces.size()) {
410  return;
411  }
412 
413  if(!(world->time % world->ticks_per_month)) {
414  if(nation->ai_do_policies) {
415  // Do a policy reform every 6 months
416  if(!((world->time / world->ticks_per_month) % 180)) {
417  // Update our policies and laws
418  ai_reform(nation);
419  }
420  }
421 
422  // Update relations with other nations
423  if(nation->ai_do_diplomacy) {
424  for(auto& other : world->nations) {
425  if(!other->exists() || other == nation) {
426  continue;
427  }
428 
429  ai_update_relations(nation, other);
430  }
431  }
432 
433  // Research technologies
434  if(nation->ai_do_research) {
435  for(auto& tech : world->technologies) {
436  // Do not research if already been completed
437  if(!nation->research[world->get_id(tech)]) {
438  continue;
439  }
440 
441  // Must be able to research it
442  if(!nation->can_research(tech)) {
443  continue;
444  }
445 
446  nation->change_research_focus(tech);
447  print_info("[%s] now researching [%s] - %.2f research points (+%.2f)", nation->ref_name.c_str(), tech->ref_name.c_str(), nation->research[world->get_id(tech)], nation->get_research_points());
448  break;
449  }
450  }
451 
452  // Accepting/rejecting treaties
453  if(nation->ai_handle_treaties) {
454  for(auto& treaty : world->treaties) {
455  for(auto& part : treaty->approval_status) {
456  if(part.first == nation) {
457  if(part.second == TreatyApproval::ACCEPTED || part.second == TreatyApproval::DENIED) break;
458 
459  if(std::rand() % 50 >= 25) {
460  print_info("We, [%s], deny the treaty of [%s]", nation->ref_name.c_str(), treaty->name.c_str());
461  part.second = TreatyApproval::DENIED;
462  }
463  else {
464  print_info("We, [%s], accept the treaty of [%s]", nation->ref_name.c_str(), treaty->name.c_str());
465  part.second = TreatyApproval::ACCEPTED;
466  }
467  }
468  }
469  }
470  }
471 
472  // Taking events
473  if(nation->ai_handle_events) {
474  for(const auto& event : nation->inbox) {
475  event->take_descision(*nation, event->descisions[std::rand() % event->descisions.size()]);
476  }
477  }
478 
479  // Build a commercially related building
480  if(!(std::rand() % 2) && nation->ai_do_build_production) {
481  ai_build_commercial(nation, world);
482  }
483 
484  if(nation->ai_do_unit_production) {
485  // Risk of invasion
486  unsigned int defense_factor = 1;
487 
488  for(const auto& rel : nation->relations) {
489  if(rel.has_war) {
490  defense_factor += 500;
491  }
492  }
493 
494  // Build defenses
495  if(!(std::rand() % std::max<int>(10, 100000 / defense_factor))) {
496  auto it = std::begin(nation->owned_provinces);
497  std::advance(it, std::rand() % nation->owned_provinces.size());
498 
499  Province* province = *it;
500  if(province->min_x > world->width || province->min_y == world->height || province->max_x < province->min_x || province->max_y < province->min_y || !province->n_tiles) {
501  print_error("Cant build defense, province doesn't have any tiles");
502  }
503  else {
504  Building* building = new Building();
505  building->province = province;
506  building->type = world->building_types[0];
507  building->owner = nation;
508  building->budget = 100.f;
509  world->insert(building);
510 
511  // Broadcast the addition of the building to the clients
512  g_server->broadcast(Action::BuildingAdd::form_packet(building));
513  print_info("Building of %s(%i), from %s built on %s", building->type->name.c_str(), (int)world->get_id(building->type), nation->name.c_str(), building->get_province()->name.c_str());
514  }
515  }
516 
517  // Build units inside buildings that are not doing anything
518  for(auto& building : g_world->buildings) {
519  if(building->working_unit_type != nullptr || building->owner != nation) {
520  continue;
521  }
522 
523  Province* province = building->get_province();
524  if(province == nullptr) {
525  continue;
526  }
527 
528  if(std::rand() % std::max<int>(10, 10000 / defense_factor)) {
529  continue;
530  }
531 
532  auto* unit_type = g_world->unit_types[std::rand() % g_world->unit_types.size()];
533  //if(!unit_type->is_ground) continue;
534  unit_type = g_world->unit_types[0];
535 
536  building->working_unit_type = unit_type;
537  building->req_goods_for_unit = unit_type->req_goods;
538  print_info("%s: Building unit [%s] in [%s]", nation->ref_name.c_str(), building->working_unit_type->ref_name.c_str(), province->ref_name.c_str());
539  }
540  }
541 
542  // Colonize a province
543  if(!(std::rand() % 50)) {
544  // Pair denoting the weight a province has, the more the better
545  std::vector<std::pair<Province*, float>> colonial_value;
546  for(const auto& province : world->provinces) {
547  if(province->owner != nullptr) {
548  continue;
549  }
550 
551  colonial_value.push_back(std::make_pair(province, 0.f));
552  }
553 
554  for(auto& prov : colonial_value) {
555  // The more population the better
556  prov.second += prov.first->total_pops() * 0.005f;
557 
558  // If it's a nucleus province it also gets a x100 multiplier to maximize priority
559  // on the nuclei provinces
560  if(prov.first->nuclei.find(nation) != prov.first->nuclei.end()) {
561  prov.second *= 100.f;
562  }
563  }
564 
565  // Found an appropriate colony!
566  if(!colonial_value.empty()) {
567  Province* target = (*std::max_element(colonial_value.begin(), colonial_value.end())).first;
568  if(target->owner == nullptr) {
569  UnifiedRender::Networking::Packet packet = UnifiedRender::Networking::Packet();
570  Archive ar = Archive();
571  ActionType action = ActionType::PROVINCE_COLONIZE;
572  ::serialize(ar, &action);
573  ::serialize(ar, &target);
574  ::serialize(ar, target);
575  packet.data(ar.get_buffer(), ar.size());
576  g_server->broadcast(packet);
577 
578  nation->give_province(*target);
579  print_info("Conquering %s for %s", target->name.c_str(), nation->name.c_str());
580  }
581  }
582  }
583  }
584 
585  // TODO: make a better algorithm
586  if(nation->ai_do_cmd_troops) {
587  std::vector<int> nations_risk_factor(world->nations.size(), 0);
588  for(const auto& other : world->nations) {
589  if(other == nation) {
590  continue;
591  }
592 
593  // Here we calculate the risk factor of each nation and then we put it on a lookup table
594  // because we can't afford to calculate this for EVERY FUCKING province
595  NationRelation& relation = nation->relations[world->get_id(other)];
596 
597  // Risk is augmentated when we border any non-ally nation
598  if(!relation.has_alliance) {
599  nations_risk_factor[world->get_id(other)] += 1;
600  }
601  if(relation.has_war) {
602  nations_risk_factor[world->get_id(other)] += 5;
603  }
604  }
605  // Our own nation is safe, let's set it to -10
606  nations_risk_factor[world->get_id(nation)] = -10;
607 
608  std::vector<int> potential_risk(world->provinces.size(), 0);
609  for(const auto& province : nation->owned_provinces) {
610  for(const auto& neighbour : province->neighbours) {
611  if(province->terrain_type->is_water_body) {
612  continue;
613  }
614 
615  // Province must be controlled by someone/not by us
616  if(neighbour->controller == nullptr || neighbour->controller == nation) {
617  continue;
618  }
619 
620  potential_risk[world->get_id(neighbour)] += nations_risk_factor[world->get_id(neighbour->controller)];
