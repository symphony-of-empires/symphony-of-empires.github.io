51 
52  std::vector<UI::ChartData> nations_data;
53  for(const auto& nation : o.gs.world->nations) {

55  }
56 
57  // Account for products that are based on this good

60  continue;
61  }
62

66  });
67

160  this->avg_price_chart->right_side_of(*this->sellers_pie);
161  this->avg_price_chart->on_each_tick = ([](UI::Widget& w, void*) {
162  auto& o = static_cast<GoodView&>(*w.parent);
163 
164  if(o.gs.world->time % o.gs.world->ticks_per_month) {
165  return;
166  }
