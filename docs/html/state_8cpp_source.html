166 void UnifiedRender::State::mixaudio(void* userdata, uint8_t* stream, int len) {
167  UnifiedRender::State& gs = *((UnifiedRender::State*)userdata);
168  std::memset(stream, 0, len);
169 
170  if(gs.sound_lock.try_lock()) {
171  for(unsigned int i = 0; i < gs.sound_queue.size(); ) {
172  int size = gs.sound_queue.size();
173  UnifiedRender::Audio* sound = gs.sound_queue[i];
174  int amount = sound->len - sound->pos;
175 
176  if(amount > len) {
177  amount = len;
178  }
179 
180  if(amount <= 0) {
181  delete sound;
182  gs.sound_queue.erase(gs.sound_queue.begin() + i);
183  continue;
184  }
185 
186  float volume = (100.f / SDL_MIX_MAXVOLUME) * gs.sound_volume;
187  SDL_MixAudio(stream, &sound->data[sound->pos], amount, SDL_MIX_MAXVOLUME - volume);
188  sound->pos += amount;
189  i++;
190  }
191 
192  for(unsigned int i = 0; i < gs.music_queue.size(); ) {
193  UnifiedRender::Audio* music = gs.music_queue[i];
194  int amount = music->len - music->pos;
195 
196  if(amount > len) {
197  amount = len;
198  }
199 
200  if(amount <= 0) {
201  delete music;
202  gs.music_queue.erase(gs.music_queue.begin() + i);
203  continue;
204  }
205 
206  float volume = (100.f / SDL_MIX_MAXVOLUME) * gs.music_volume;
207  SDL_MixAudio(stream, &music->data[music->pos], amount, (SDL_MIX_MAXVOLUME / gs.music_fade_value) - volume);
208  music->pos += amount;
209  i++;
210  }
211  gs.sound_lock.unlock();
212  }
213 
214  if(gs.music_fade_value > 1.f) {
215  gs.music_fade_value -= 1.f;
216  }
217 }
