100  bool get_cursor_map_pos(std::pair<int, int> mouse_pos, glm::ivec2& out_pos) override {
101  float mouse_x = mouse_pos.first;
102  float mouse_y = screen_size.y - 1.f - mouse_pos.second;
103 
104  glm::mat4 view = get_view();
105  glm::mat4 projection = get_projection();
106 
107  glm::vec3 world_space_near = glm::unProject(
108  glm::vec3(mouse_x, mouse_y, 0.f),
109  view, projection,
110  glm::vec4(0.f, 0.f, screen_size));
111 
112  glm::vec3 world_space_far = glm::unProject(
113  glm::vec3(mouse_x, mouse_y, 1.f),
114  view, projection,
115  glm::vec4(0.f, 0.f, screen_size)
116  );
117 
118  glm::vec3 ray_direction = world_space_far - world_space_near;
119 
120  float distance = 0.f;
121  bool hit = glm::intersectRayPlane(world_space_near, ray_direction, glm::vec3(0, 0, 0), glm::vec3(0, 0, 1), distance);
122 
123  glm::vec3 intersection_point = world_space_near + ray_direction * distance;
124  out_pos.x = intersection_point.x;
125  out_pos.y = intersection_point.y;
126  if(intersection_point.x < 0
127  || intersection_point.y < 0
128  || intersection_point.x > map_size.x
129  || intersection_point.y > map_size.y) {
130  hit = false;
131  }
132  return hit;
133  };
