163 std::vector<ProvinceColor> population_map_mode(const World& world) {
164  // Find the maximum amount of pops in one province
165  std::vector<std::pair<Province::Id, float>> province_amounts;
166  float max_amount = 1;
167  for(auto const& province : world.provinces) {
168  float amount = 0;
169  for(auto const& pop : province->pops) {
170  amount += pop.size;
171  }
172  amount = log2(amount + 1.f);
173  max_amount = std::max<float>(amount, max_amount);
174  province_amounts.push_back(std::make_pair(world.get_id(province), amount));
175  }
176 
177  // Mix each color depending of how many live there compared to max_amount
178  UnifiedRender::Color min = UnifiedRender::Color::rgb8(255, 229, 217);
179  UnifiedRender::Color max = UnifiedRender::Color::rgb8(220, 46, 35);
180  std::vector<ProvinceColor> province_color;
181  for(auto const& prov_amount : province_amounts) {
182  Province::Id prov_id = prov_amount.first;
183  uint32_t amount = prov_amount.second;
184  float ratio = amount / max_amount;
185  UnifiedRender::Color color = UnifiedRender::Color::lerp(min, max, ratio);
186  province_color.push_back(ProvinceColor(prov_id, color));
187  }
188  return province_color;
189 }

204 std::vector<ProvinceColor> culture_map_mode(const World& world) {
205  std::vector<ProvinceColor> province_color;
206  UnifiedRender::Color min = UnifiedRender::Color::rgb8(255, 255, 255);
207  for(unsigned int i = 0; i < world.provinces.size(); i++) {
208  Province* province = world.provinces[i];
209  std::unordered_map<Culture::Id, size_t> culture_amounts;
210  size_t total_amount = 0;
211  size_t max_amount = 0;
212  Culture::Id max_culture_id = 0;
213  for(unsigned int j = 0; j < province->pops.size(); j++) {
214  const Pop& pop = province->pops[j];
215  total_amount += pop.size;
216 
217  auto search = culture_amounts.find(pop.culture->cached_id);
218  if(search == culture_amounts.end()) {
219  culture_amounts[pop.culture->cached_id] = pop.size;
220  }
221  else {
222  culture_amounts[pop.culture->cached_id] += pop.size;
223  }
224  size_t amount = culture_amounts[pop.culture->cached_id];
225  if(amount > max_amount) {
226  max_amount = amount;
227  max_culture_id = pop.culture->cached_id;
228  }
229  }
230  UnifiedRender::Color max = UnifiedRender::Color::rgba32(world.cultures[max_culture_id]->color);
231  UnifiedRender::Color color = UnifiedRender::Color::lerp(min, max, ((float)max_amount) / total_amount);
232  province_color.push_back(ProvinceColor(i, color));
233  }
234  // Water
235  province_color.push_back(ProvinceColor((Province::Id)-2, UnifiedRender::Color::rgba32(0x00000000)));
236  // Land
237  province_color.push_back(ProvinceColor((Province::Id)-1, UnifiedRender::Color::rgba32(0xffdddddd)));
238  return province_color;
239 }

241 std::string culture_tooltip(const World& world, const Province::Id id){
242  Province* province = world.provinces[id];
243 
244  typedef std::pair<Culture::Id, size_t> culture_amount;
245  std::vector<culture_amount> cultures;
246  for(unsigned int i = 0; i < province->pops.size(); i++) {
247  const Pop& pop = province->pops[i];
248 
249  bool found = false;
250  for(auto culture_amount : cultures) {
251  if(culture_amount.first == pop.culture->cached_id) {
252  culture_amount.second += pop.size;
253  found = true;
254  }
255  }
256  if(!found) {
257  cultures.push_back(std::make_pair(pop.culture->cached_id, pop.size));
258  }
259  }
260  std::sort(cultures.begin(), cultures.end(), [](culture_amount a, culture_amount b) {
261  return a.second > b.second;
262  });
263  std::string out;
264  for(auto culture_amount : cultures) {
265  out += world.cultures[culture_amount.first]->name + std::to_string(culture_amount.second) + "\n";
266  }
267  return out;
