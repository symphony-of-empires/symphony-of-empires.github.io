75 BuildingSelectTypeTab::BuildingSelectTypeTab(GameState& _gs, int x, int y, UI::Widget* _parent)
76  : UI::Group(x, y, _parent->width - x, _parent->height - y, _parent),
77  gs{ _gs }
78 {
79  unsigned int i = 0;
80  for(const auto& building_type : gs.world->building_types) {
81  auto* btn = new BuildingTypeButton(gs, 0, 24 * i, building_type, this);
82  btn->on_click = ([](UI::Widget& w, void*) {
83  auto& o = static_cast<BuildingBuildView&>(*w.parent->parent);
84  if(o.nation == nullptr) {
85  o.gs.ui_ctx->prompt("Error", "No nation selected");
86  return;
87  }
88 
89  if(o.province == nullptr) {
90  o.gs.ui_ctx->prompt("Error", "Select a province to build it on");
91  return;
92  }
93 
94  if(o.gs.curr_nation->owned_provinces.empty()) {
95  o.gs.ui_ctx->prompt("Error", "You do not own any provinces");
96  return;
97  }
98 
99  auto building = Building();
100  building.owner = o.nation;
101  building.type = ((BuildingTypeButton&)w).building_type;
102  building.province = o.province;
103  g_client->send(Action::BuildingAdd::form_packet(&building));
104 
105  o.gs.ui_ctx->prompt("Production", "Building a " + building.type->name + " in " + building.get_province()->name + "; owned by " + building.get_owner()->name);
106  });
107  i++;
108  }
109 }
