33 // Construct a shader by opening the provided path and creating a temporal ifstream, reading
34 // from that stream in text mode and then compiling the shader
35 UnifiedRender::OpenGL::Shader::Shader(const std::string& path, GLuint type, bool use_transpiler, std::vector<UnifiedRender::OpenGL::GLSL_Define> defintions) {
36  std::ifstream file;
37  file.exceptions(std::ifstream::failbit | std::ifstream::badbit);
38  try {
39  // Read the entire file onto the stringstream
40  file.open(Path::get(path).c_str());
41  std::stringstream stream;
42  stream << file.rdbuf();
43  file.close();
44 
45  buffer = stream.str();
46 
47  if(use_transpiler) {
48  UnifiedRender::OpenGL::GLSL_Context ctx(buffer);
49  ctx.defines = defintions;
50  ctx.lexer();
51  try {
52  ctx.parser();
53  } catch(UnifiedRender::OpenGL::GLSL_Exception& e) {
54  print_error("%s -> %s", e.it->data.c_str(), e.what());
55  }
56  buffer = ctx.to_text();
57  }
58 
59  const char* c_code = buffer.c_str();
60  id = glCreateShader(type);
61  glShaderSource(id, 1, &c_code, NULL);
62 
63  compile(type);
64  } catch(std::ifstream::failure& e) {
65  print_error("Cannot load shader %s", path.c_str());
66  }
67 }

70 void UnifiedRender::OpenGL::Shader::compile(GLuint type) {
71  glCompileShader(id);
72 
73  // Check for errors of the shader
74  GLint r = 0;
75  glGetShaderiv(id, GL_COMPILE_STATUS, &r);
76  if(!r) {
77  std::string error_info;
78  glGetShaderInfoLog(id, GL_INFO_LOG_LENGTH, NULL, &error_info[0]);
79  print_error("Raw status: %s", error_info.c_str());
80 
81  // Nvidia's style of errors
82  std::istringstream sline(error_info);
83  int slot, row, col;
84  char ch;
85 
86  // nvidia styles errors the following way:
87  // slot:row(column)
88  // They also like to fuck me over, so they use
89  // slot(row) <-- fuck this
90 
91  sline >> slot >> ch;
92  if(ch == ':') {
93  sline >> row >> ch;
94  if(ch == '(') {
95  sline >> col;
96  }
97  } else if(ch == '(') {
98  sline >> row;
99  }
100 
101  // Now we will pray that this actually worked
102 
103  // Next we will read over the shader file and find the line
104  int row_rem = row;
105  std::string::iterator it;
106  for(it = buffer.begin(); it != buffer.end() && row_rem; it++) {
107  if(*it == '\n') {
108  row_rem--;
109  if(!row_rem) {
110  break;
111  }
112  }
113  }
114 
115  std::string line_buf = "(No line info)";
116  if(it != buffer.end()) {
117  line_buf = buffer.substr(std::distance(buffer.begin(), it), buffer.find_first_of('\n', std::distance(buffer.begin(), it)));
118  }
119  throw ShaderException(line_buf + "\n" + error_info);
120  }
121  // print_info("Status: Sucess");
122 }

210 UnifiedRender::OpenGL::Program::Program(const UnifiedRender::OpenGL::VertexShader* vertex, const UnifiedRender::OpenGL::FragmentShader* fragment, const UnifiedRender::OpenGL::GeometryShader* geometry, const UnifiedRender::OpenGL::TessControlShader* tctrl, const UnifiedRender::OpenGL::TessEvalShader* tee) {
211  id = glCreateProgram();
212  glBindAttribLocation(id, 0, "m_pos");
213  glBindAttribLocation(id, 1, "m_texcoord");
214 #ifdef UR_RENDER_DEBUG
215  if(vertex == nullptr || !vertex->get_id()) {
216  throw UnifiedRender::DebugException("Vertex shader object was not provided correctly");
217  }
218 #endif
219  attach_shader(vertex);
220 
221 #ifdef UR_RENDER_DEBUG
222  if(fragment == nullptr || !fragment->get_id()) {
223  throw UnifiedRender::DebugException("Vertex shader object was not provided correctly");
224  }
225 #endif
226  attach_shader(fragment);
227 
228  if(geometry != nullptr) {
229  attach_shader(geometry);
230  }
231 
232  if(tctrl != nullptr) {
233  attach_shader(tctrl);
234  }
235 
236  if(tee != nullptr) {
237  attach_shader(tee);
238  }
239  link();
240 }

257 std::unique_ptr<UnifiedRender::OpenGL::Program> UnifiedRender::OpenGL::Program::create(std::vector<UnifiedRender::OpenGL::Option> options, const std::string& vs_path, const std::string& fs_path, const std::string& gs_path) {
