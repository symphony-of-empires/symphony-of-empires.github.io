131  template<typename T>
132  inline void recv(T* buf = nullptr) {
133  uint32_t net_code;
134  stream.recv(&net_code, sizeof(net_code));
135  net_code = ntohl(net_code);
136  code = static_cast<PacketCode>(net_code);
137 
138  uint32_t net_size;
139  stream.recv(&net_size, sizeof(net_size));
140  n_data = (size_t)ntohl(net_size);
141  buffer.resize(n_data + 1);
142 
143  // Reads can only be done 1024 bytes at a time
144  stream.recv(&buffer[0], n_data);
145  if(buf != nullptr) {
146  //size_t new_size = get_compressed_len(n_data);
147  //char* new_buf = new char[new_size];
148  //size_t decomp_size = decompress(buf, n_data, new_buf, new_size);
149  //std::memcpy(buf, &new_buf[0], decomp_size);
150  //delete[] new_buf;
151  std::memcpy(buf, &buffer[0], n_data);
152  }
153 
154  uint16_t eof_marker;
155  stream.recv(&eof_marker, sizeof(eof_marker));
156  if(ntohs(eof_marker) != 0xE0F) {
157  throw UnifiedRender::Networking::SocketException("Packet with invalid EOF");
158  }
159  }
