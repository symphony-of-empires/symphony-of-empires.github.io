129 // The standard map mode with each province color = country color
130 std::vector<ProvinceColor> political_map_mode(const World& world) {
131  std::vector<ProvinceColor> province_color;
132  for(unsigned int i = 0; i < world.provinces.size(); i++) {
133  Nation* province_owner = world.provinces[i]->owner;
134  if(province_owner == nullptr) {
135  province_color.push_back(ProvinceColor(i, UnifiedRender::Color::rgba32(0xffdddddd)));
136  }
137  else if(province_owner->cached_id == (Nation::Id)-1) {
138  province_color.push_back(ProvinceColor(i, UnifiedRender::Color::rgba32(0xffdddddd)));
139  }
140  else {
141  province_color.push_back(ProvinceColor(i, UnifiedRender::Color::rgba32(province_owner->get_client_hint().color)));
142  }
143  }
144 
145  // Water
146  province_color.push_back(ProvinceColor((Province::Id)-2, UnifiedRender::Color::rgba32(0x00000000)));
147  // Land
148  province_color.push_back(ProvinceColor((Province::Id)-1, UnifiedRender::Color::rgba32(0xffdddddd)));
149  return province_color;
150 }

169 void Map::draw_flag(const UnifiedRender::OpenGL::Program& shader, const Nation& nation) {
170  // Draw a flag that "waves" with some cheap wind effects it
171  // looks nice and it's super cheap to make - only using sine
172  const float n_steps = 8.f; // Resolution of flag in one side (in vertices)
173  const float step = 90.f; // Steps per vertice
174 
175  auto flag = UnifiedRender::Mesh<glm::vec3, glm::vec2>(UnifiedRender::MeshMode::TRIANGLE_STRIP);
176  for(float r = 0.f; r <= (n_steps * step); r += step) {
177  float sin_r = (sin(r + wind_osc) / 24.f);
178 
179  sin_r = sin(r + wind_osc) / 24.f;
180  flag.buffer.push_back(UnifiedRender::MeshData<glm::vec3, glm::vec2>(
181  glm::vec3(((r / step) / n_steps) * 1.5f, sin_r, -2.f),
182  glm::vec2((r / step) / n_steps, 0.f)
183  ));
184 
185  sin_r = sin(r + wind_osc + 160.f) / 24.f;
186  flag.buffer.push_back(UnifiedRender::MeshData<glm::vec3, glm::vec2>(
187  glm::vec3(((r / step) / n_steps) * 1.5f, sin_r, -1.f),
188  glm::vec2((r / step) / n_steps, 1.f)
189  ));
190  }
191  flag.upload();
192 
193  shader.set_texture(0, "diffuse_map", *nation_flags[world.get_id(&nation)]);
194  flag.draw();
195 }

197 void Map::handle_click(GameState& gs, SDL_Event event) {
198  Input& input = gs.input;
199 
200  if(input.select_pos.first < 0 || input.select_pos.first >= gs.world->width || input.select_pos.second < 0 || input.select_pos.second >= gs.world->height) {
201  return;
202  }
203 
204  if(event.button.button == SDL_BUTTON_LEFT) {
205  std::pair<float, float>& select_pos = input.select_pos;
206  const Tile& tile = gs.world->get_tile(select_pos.first, select_pos.second);
207  switch(gs.current_mode) {
208  case MapMode::COUNTRY_SELECT:
209  if(tile.province_id < (Province::Id)-3) {
210  auto province = world.provinces[tile.province_id];
211  if(province->controller == nullptr) {
212  break;
213  }
214  gs.select_nation->change_nation(province->controller->cached_id);
215  }
216  break;
217  case MapMode::NORMAL:
218  // Check if we selected an unit
219  input.selected_units.clear();
220  for(const auto& unit : gs.world->units) {
221  // We can't control others units
222  if(unit->owner != gs.curr_nation) {
223  continue;
224  }
225 
226  std::pair<float, float> pos = unit->get_pos();
227 
228  std::pair<float, float> start_pos = input.drag_coord;
229  if(input.select_pos.first < input.drag_coord.first) {
230  start_pos.first = input.select_pos.first;
231  } if(input.select_pos.second < input.drag_coord.second) {
232  start_pos.second = input.select_pos.second;
233  }
234 
235  std::pair<float, float> end_pos = input.drag_coord;
236  if(input.select_pos.first > input.drag_coord.first) {
237  end_pos.first = input.select_pos.first;
238  } if(input.select_pos.second > input.drag_coord.second) {
239  end_pos.second = input.select_pos.second;
240  }
241 
242  if(pos.first >= start_pos.first && pos.first <= end_pos.first && pos.second >= start_pos.second && pos.second <= end_pos.second) {
243  input.selected_units.push_back(unit);
244  }
245  }
246  is_drag = false;
247 
248  if(input.selected_units.empty()) {
249  // Show province information when clicking on a province
250  if(tile.province_id < gs.world->provinces.size()) {
251  new Interface::ProvinceView(gs, gs.world->provinces[tile.province_id]);
252  return;
253  }
254  }
255  break;
256  default:
257  break;
258  }
259  return;
260  }
261  else if(event.button.button == SDL_BUTTON_RIGHT) {
262  const Tile& tile = gs.world->get_tile(input.select_pos.first, input.select_pos.second);
263  if(tile.province_id == (Province::Id)-1) {
264  return;
265  }
266 
267  Province* province = gs.world->provinces[tile.province_id];
268 
269  //if(input.selected_units.empty()) {
270  if(gs.editor) {
271  gs.curr_nation->give_province(*province);
272  province->nuclei.insert(gs.curr_nation);
273  update_mapmode();
274  }
275 
276  for(const auto& unit : input.selected_units) {
277  if(!unit->province->is_neighbour(*province)) {
278  continue;
279  }
280 
281  if(unit->province->controller != nullptr && unit->province->controller != gs.curr_nation) {
282  // Must either be our ally, have military access with them or be at war
283  const NationRelation& relation = gs.curr_nation->relations[gs.world->get_id(unit->province->controller)];
284  if(!(relation.has_war || relation.has_alliance || relation.has_military_access)) {
285  continue;
286  }
287  }
288 
289  if(!unit->can_move()) {
290  continue;
291  }
292 
293  UnifiedRender::Networking::Packet packet = UnifiedRender::Networking::Packet();
294  Archive ar = Archive();
295  ActionType action = ActionType::UNIT_CHANGE_TARGET;
296  ::serialize(ar, &action);
297  ::serialize(ar, &unit);
298  ::serialize(ar, &province);
299  packet.data(ar.get_buffer(), ar.size());
300  std::scoped_lock lock(gs.client->pending_packets_mutex);
301  gs.client->pending_packets.push_back(packet);
302 
303  std::scoped_lock l1(gs.sound_lock);
304  auto entries = Path::get_all_recursive("sfx/land_move");
305  if(!entries.empty()) {
306  gs.sound_queue.push_back(new UnifiedRender::Audio(entries[std::rand() % entries.size()]));
307  }
308  }
309  }
310 }

312 void Map::update(const SDL_Event& event, Input& input, UI::Context* ui_ctx) {
313  std::pair<int, int>& mouse_pos = input.mouse_pos;
314  // std::pair<float, float>& select_pos = input.select_pos;
315  switch(event.type) {
316  case SDL_JOYBUTTONDOWN:
317  case SDL_MOUSEBUTTONDOWN:
318  SDL_GetMouseState(&mouse_pos.first, &mouse_pos.second);
319 
320  is_drag = false;
321  if(event.button.button == SDL_BUTTON_MIDDLE) {
322  glm::ivec2 map_pos;
323  if(camera->get_cursor_map_pos(input.mouse_pos, map_pos)) {
324  last_camera_drag_pos = map_pos;
325  input.last_camera_mouse_pos = mouse_pos;
326  }
327  }
328  else if(event.button.button == SDL_BUTTON_LEFT) {
329  input.drag_coord = input.select_pos;
330  input.drag_coord.first = (int)input.drag_coord.first;
331  input.drag_coord.second = (int)input.drag_coord.second;
332  is_drag = true;
333  }
334  break;
335  case SDL_JOYBUTTONUP:
336  case SDL_MOUSEBUTTONUP:
337  is_drag = false;
338  break;
339  case SDL_JOYAXISMOTION:{
340  int xrel = SDL_JoystickGetAxis(input.joy, 0);
341  int yrel = SDL_JoystickGetAxis(input.joy, 1);
342 
343  const float sensivity = UnifiedRender::State::get_instance().joy_sensivity;
344 
345  float x_force = xrel / sensivity;
346  float y_force = yrel / sensivity;
347 
348  input.mouse_pos.first += x_force;
349  input.mouse_pos.second += y_force;
350 
351  if(input.middle_mouse_down) { // Drag the map with middlemouse
352  glm::ivec2 map_pos;
353  if(camera->get_cursor_map_pos(input.mouse_pos, map_pos)) {
354  glm::vec2 current_pos = glm::make_vec2(camera->get_map_pos());
355  const glm::vec2 pos = current_pos + last_camera_drag_pos - glm::vec2(map_pos);
356  camera->set_pos(pos.x, pos.y);
357  }
358  }
359  glm::ivec2 map_pos;
360  if(camera->get_cursor_map_pos(input.mouse_pos, map_pos)) {
361  input.select_pos.first = map_pos.x;
362  input.select_pos.second = map_pos.y;
363  }
364  } break;
365  case SDL_MOUSEMOTION:
366  SDL_GetMouseState(&mouse_pos.first, &mouse_pos.second);
367  glm::ivec2 map_pos;
368  if(input.middle_mouse_down) { // Drag the map with middlemouse
369  if(camera->get_cursor_map_pos(mouse_pos, map_pos)) {
370  glm::vec2 current_pos = glm::make_vec2(camera->get_map_pos());
371  const glm::vec2 pos = current_pos + last_camera_drag_pos - glm::vec2(map_pos);
372  camera->set_pos(pos.x, pos.y);
373  }
374  }
375  if(camera->get_cursor_map_pos(mouse_pos, map_pos)) {
376  input.select_pos.first = map_pos.x;
377  input.select_pos.second = map_pos.y;
378  auto prov_id = world.get_tile(map_pos.x, map_pos.y).province_id;
379  if(mapmode_tooltip_func != nullptr) {
380  std::string text = mapmode_tooltip_func(world, prov_id);
381  UI::Tooltip* tooltip = new UI::Tooltip();
382  tooltip->text(text);
383  ui_ctx->use_tooltip(tooltip, glm::ivec2(mouse_pos.first, mouse_pos.second));
384  }
385  }
386  break;
387  case SDL_MOUSEWHEEL:
388  SDL_GetMouseState(&mouse_pos.first, &mouse_pos.second);
389  camera->move(0.f, 0.f, event.wheel.y * -2.f);
390  break;
391  case SDL_KEYDOWN:
392  switch(event.key.keysym.sym) {
393  case SDLK_UP:
394  camera->move(0.f, -1.f, 0.f);
395  break;
396  case SDLK_DOWN:
397  camera->move(0.f, 1.f, 0.f);
398  break;
399  case SDLK_LEFT:
400  camera->move(-1.f, 0.f, 0.f);
401  break;
402  case SDLK_RIGHT:
403  camera->move(1.f, 0.f, 0.f);
404  break;
405  }
406  break;
407  }
408 }
