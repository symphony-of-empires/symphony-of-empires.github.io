177  template<typename T>
178  inline void insert(T* ptr) {

181  list.push_back(ptr);

184  template<typename T>
185  inline void remove(T* ptr) {
186  // Decrease the cache_id counter for the elements after the removed element
187  typename T::Id cached_id = this->get_id<T>(ptr);
188  auto& list = this->get_list(ptr);
189  for(typename T::Id i = cached_id + 1; i < list.size(); i++) {
190  list[i]->cached_id--;
191  }
192 
193  // Remove the element itself
194  list.erase(list.begin() + cached_id);
195  };
