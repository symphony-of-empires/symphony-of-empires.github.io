113 UnifiedRender::MeshData<glm::vec3, glm::vec2> UnifiedRender::Sphere::calc_pos(glm::vec3 center_pos, float longitude, float latitude) {
114  float longitude_ratio = ((float)longitude) / resolution;
115  float longitude_rad = longitude_ratio * 2 * M_PI;
116  float latitude_ratio = ((float)latitude) / resolution;
117  float latitude_rad = latitude_ratio * M_PI;
118  float x = radius * std::cos(longitude_rad) * std::sin(latitude_rad);
119  float y = radius * std::sin(longitude_rad) * std::sin(latitude_rad);
120  float z = radius * std::cos(latitude_rad);
121  glm::vec3 pos(x, y, z);
122  pos += center_pos;
123  glm::vec2 tex_coord(longitude_ratio, latitude_ratio);
124  return UnifiedRender::MeshData<glm::vec3, glm::vec2>(pos, tex_coord);
125 }
