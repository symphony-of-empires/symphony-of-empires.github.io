124 // Change nation in start screen
125 void LobbySelectView::change_nation(size_t id) {
126  size_t old_id = curr_selected_nation;
127  if (!g_world->nations.size()) {
128  gs.ui_ctx->prompt("Error", "No nations to select");
129  return;
130  }
131 
132  if (id >= g_world->nations.size()) {
133  id = 0;
134  }
135 
136  gs.curr_nation = gs.world->nations[id];
137  if (id > old_id) {
138  while (gs.curr_nation->exists() == false) {
139  id++;
140  if (id >= gs.world->nations.size()) {
141  id = 0;
142  }
143  gs.curr_nation = gs.world->nations[id];
144  }
145  } else {
146  while (gs.curr_nation->exists() == false) {
147  id--;
148  if (id >= gs.world->nations.size()) {
149  id = gs.world->nations.size() - 1;
150  }
151  gs.curr_nation = gs.world->nations[id];
152  }
153  }
154 
155  curr_selected_nation = id;
156  curr_country_btn->text(gs.curr_nation->name.c_str());
157  const Province* capital = gs.curr_nation->capital;
158  if(capital != nullptr) {
159  gs.map->camera->set_pos(capital->max_x, capital->max_y);
160  }
161 }
