50 UnifiedRender::Audio::Audio(const std::string& path) {
51  SDL_AudioSpec wave;
52  SDL_AudioCVT cvt;
53 
54  int channels, rate;
55  uint8_t* decoded;
56  this->len = stb_vorbis_decode_filename(path.c_str(), &channels, &rate, (short**)&decoded);
57  if(!this->len) {
58  throw UnifiedRender::AudioException(path, "0 length audio");
59  }
60  this->len = this->len * channels * (sizeof(int16_t) / sizeof(uint8_t));
61  this->data = decoded;
62  this->pos = 0;
63 
64  // stb already loads OGG as a series of U16 nodes, so we only have to use AUDIO_S16
65  // and the rest is already given by stb
66  SDL_BuildAudioCVT(&cvt, AUDIO_S16, channels, rate, AUDIO_S16, 1, 11050);
67  cvt.buf = (Uint8*)malloc(this->len * cvt.len_mult);
68  if(cvt.buf == nullptr) {
69  throw UnifiedRender::AudioException(path, "Cannot allocate memory");
70  }
71  std::memcpy(cvt.buf, this->data, this->len);
72  cvt.len = this->len;
73  SDL_ConvertAudio(&cvt);
74 
75  this->data = cvt.buf;
76  this->len = cvt.len_cvt;
77  this->pos = 0;
78 }
