33 std::string treaty_to_text(Treaty* treaty) {
34  std::string text = "";
35  for(const auto& clause : treaty->clauses) {
36  if(clause->type == TreatyClauseType::WAR_REPARATIONS) {
37  const auto* dyn_clause = static_cast<const TreatyClause::WarReparations*>(clause);
38  text += dyn_clause->sender->name + " demands $" + std::to_string(dyn_clause->amount) + " from " + dyn_clause->receiver->name;
39  } else if(clause->type == TreatyClauseType::ANEXX_PROVINCES) {
40  const auto* dyn_clause = static_cast<const TreatyClause::AnexxProvince*>(clause);
41  text += dyn_clause->sender->name + " obtains ";
42  for(const auto& province : dyn_clause->provinces) {
43  text += province->name + " from " + dyn_clause->receiver->name + ", ";
44  }
45  } else if(clause->type == TreatyClauseType::LIBERATE_NATION) {
46  const auto* dyn_clause = static_cast<const TreatyClause::LiberateNation*>(clause);
47  text += dyn_clause->sender->name + " liberates " + dyn_clause->liberated->name + " and gives them ";
48  for(const auto& province : dyn_clause->provinces) {
49  text += province->name + " from " + dyn_clause->receiver->name + ", ";
50  }
51  } else if(clause->type == TreatyClauseType::HUMILIATE) {
52  const auto* dyn_clause = static_cast<const TreatyClause::Humiliate*>(clause);
53  text += dyn_clause->sender->name + " humiliates " + dyn_clause->receiver->name;
54  } else if(clause->type == TreatyClauseType::IMPOSE_POLICIES) {
55  const auto* dyn_clause = static_cast<const TreatyClause::ImposePolicies*>(clause);
56  text += dyn_clause->sender->name + " imposes policies on " + dyn_clause->receiver->name;
57  } else if(clause->type == TreatyClauseType::CEASEFIRE) {
58  const auto* dyn_clause = static_cast<const TreatyClause::Ceasefire*>(clause);
59  text += dyn_clause->sender->name + " proposes ceasefire to " + dyn_clause->receiver->name;
60  }
61  text += ", ";
62  }
63  return text;
64 }
