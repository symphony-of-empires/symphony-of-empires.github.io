68 void PieChart::draw_triangle(float start_ratio, float end_ratio, UnifiedRender::Color color) {
69  float x_center = width / 2.f;
70  float y_center = height / 2.f;
71  float radius = std::min<float>(width, height) * 0.5;
72  float x_offset, y_offset, scale;
73 
74  glColor3f(color.r, color.g, color.b);
75  x_offset = cos((start_ratio - 0.25f) * 2 * M_PI);
76  y_offset = sin((start_ratio - 0.25f) * 2 * M_PI);
77  scale = std::min<float>(1.f / abs(x_offset), 1.f / abs(y_offset));
78  x_offset *= scale;
79  y_offset *= scale;
80  glTexCoord2f(0.5f + x_offset * 0.5f, 0.5f + y_offset * 0.5f);
81  glVertex2f(x_center + x_offset * radius, y_center + y_offset * radius);
82 
83  x_offset = cos((end_ratio - 0.25f) * 2 * M_PI);
84  y_offset = sin((end_ratio - 0.25f) * 2 * M_PI);
85  scale = std::min<float>(1.f / abs(x_offset), 1.f / abs(y_offset));
86  x_offset *= scale;
87  y_offset *= scale;
88  glTexCoord2f(0.5f + x_offset * 0.5f, 0.5f + y_offset * 0.5f);
89  glVertex2f(x_center + x_offset * radius, y_center + y_offset * radius);

91  glTexCoord2f(0.5f, 0.5f);
92  glVertex2f(x_center, y_center);

95 void PieChart::on_render(Context& ctx, UnifiedRender::Rect viewport) {
96  glBindTexture(GL_TEXTURE_2D, ctx.piechart_overlay->gl_tex_num);
97  glBegin(GL_TRIANGLES);
98  float counter = 0;
99  float last_corner = -0.125f;
100  float last_ratio = 0;
101 
102  for(auto& slice : data) {
103  counter += slice.num;
104  float ratio = counter / max;
105  while(ratio > last_corner + 0.25f) {
106  last_corner += 0.25f;
107  draw_triangle(last_ratio, last_corner, slice.color);
108  last_ratio = last_corner;
109  }
110  draw_triangle(last_ratio, ratio, slice.color);
111  last_ratio = ratio;
112  }
113  glEnd();
114 }

116 bool in_triangle(glm::vec2 p, glm::vec2 center, float radius, float start_ratio, float end_ratio) {
117  float x_offset, y_offset, scale;
118  x_offset = cos((start_ratio - 0.25f) * 2 * M_PI);
119  y_offset = sin((start_ratio - 0.25f) * 2 * M_PI);
120  scale = std::min<float>(1.f / abs(x_offset), 1.f / abs(y_offset));
121  x_offset *= scale;
122  y_offset *= scale;
123  glm::vec2 a(center.x + x_offset * radius, center.y + y_offset * radius);
124 
125  x_offset = cos((end_ratio - 0.25f) * 2 * M_PI);
126  y_offset = sin((end_ratio - 0.25f) * 2 * M_PI);
127  scale = std::min<float>(1.f / abs(x_offset), 1.f / abs(y_offset));
128  x_offset *= scale;
129  y_offset *= scale;
130  glm::vec2 b(center.x + x_offset * radius, center.y + y_offset * radius);
131 
132  glm::vec2 c(center.x, center.y);
133 
134  float A = 0.5f * (-b.y * c.x + a.y * (-b.x + c.x) + a.x * (b.y - c.y) + b.x * c.y);
135  float sign = A < 0 ? -1 : 1;
136  float s = (a.y * c.x - a.x * c.y + (c.y - a.y) * p.x + (a.x - c.x) * p.y) * sign;
137  float t = (a.x * b.y - a.y * b.x + (a.y - b.y) * p.x + (b.x - a.x) * p.y) * sign;
138 
139  return s > 0 && t > 0 && (s + t) < 2 * A * sign;
140 }

142 void PieChart::on_hover_default(Widget& w, glm::ivec2 mouse_pos, glm::ivec2 widget_pos) {
143  PieChart& piechart = static_cast<PieChart&>(w);
144  mouse_pos -= widget_pos;
145  float counter = 0;
146  float last_corner = -0.125f;
147  float last_ratio = 0;
148 
149  float x_center = piechart.width / 2.f;
150  float y_center = piechart.height / 2.f;
151  glm::ivec2 center = glm::ivec2(x_center, y_center);
152  float radius = std::min<float>(piechart.width, piechart.height) * 0.5;
153 
154  glm::vec2 centered_pos = mouse_pos - center;
155  if(glm::length(centered_pos) > radius) {
156  piechart.tooltip->is_render = false;
157  return;
158  }
159 
160  for(auto& slice : piechart.data) {
161  counter += slice.num;
162  float ratio = counter / piechart.max;
163  while(ratio > last_corner + 0.25f) {
164  last_corner += 0.25f;
165  bool is_inside = in_triangle(mouse_pos, center, radius, last_ratio, last_corner);
166  if(is_inside) {
167  piechart.tooltip->text(slice.info);
168  piechart.tooltip->is_render = true;
169  return;
170  }
171  last_ratio = last_corner;
172  }
173  bool is_inside = in_triangle(mouse_pos, center, radius, last_ratio, last_corner);
174  if(is_inside) {
175  piechart.tooltip->text(slice.info);
176  piechart.tooltip->is_render = true;
177  return;
178  }
179  last_ratio = ratio;
180  }
181 }
