129 float Unit::get_speed(const Province& _province) const {
130  auto start_pos = province->get_pos();
131  auto end_pos = _province.get_pos();
132 
133  // Get the linear distance from the current deduced position of the unit and the target
134  // the current position of the unit is relative to the move progress it has done (so if it's
135  // halfway thru a province it will then be placed at half of the distance)
136  const float x_dist = (end_pos.first - start_pos.first);
137  const float y_dist = (end_pos.second - start_pos.second);
138  const float angle = std::atan2(x_dist, y_dist);
139 
140  // TODO: The comment above makes no sense since we don't do (max_move_progress / move_progress)
141  const float dist_div = move_progress;
142 
143  //const float linear_dist = std::fabs(std::sqrt(x_dist * x_dist + y_dist * y_dist) / dist_div);
144 
145  const float speed = type->speed / _province.terrain_type->movement_penalty;
146  float radius_scale = std::cos(M_PI / (2 * World::get_instance().height) * (2 * (y_dist / dist_div) - World::get_instance().height));
147  float x_scale = 1 / (std::fabs(radius_scale) + 0.001f);
148  float speed_scale = std::sqrt(std::pow(std::sin(angle), 2) + std::pow(std::cos(angle) * x_scale, 2));
149  return (speed * speed_scale) / 100.f;
