33 inline bool Diplomacy::is_friend(Nation& us, Nation& them) {
34  const NationRelation* relation = &us.relations[g_world->get_id(&them)];
35 
36  // A high relation means we are friendly <3
37  if(relation->relation >= 50.f) {
38  return true;
39  } else {
40  // Well, but maybe our interest is able to determine our friendliness towards them?
41  if(relation->interest >= relation->relation) {
42  // We cannot be friendly with negative relations
43  if(relation->relation <= 0.f) {
44  return false;
45  }
46 
47  // Well, we need to be interested enough to like them
48  if(relation->relation >= relation->interest / relation->relation) {
49  return true;
50  }
51  }
52  return false;
53  }
54 }

138 void AnexxProvince::enforce(void) {
139  sender->prestige += cost() * 0.0000025f;
140  receiver->prestige -= cost() * 0.000005f;
141 
142  // Give provinces to the winner
143  for(auto& province : provinces) {
144  UnifiedRender::Log::debug("game", "Giving " + province->ref_name + " to " + sender->ref_name + " from " + receiver->ref_name);
145 
146  province->owner = sender;
147 
148  // Add to sender's owned provinces list and remove from
149  // original owner
150  sender->owned_provinces.insert(province);
151  const auto it = receiver->owned_provinces.find(province);
152  if(it != receiver->owned_provinces.end()) {
153  receiver->owned_provinces.erase(it);
154  }
155  }
156 
157  // One-time clause
158  done = true;
159 }

196 // Checks if the treaty has any clause which may still make the treaty be in effect
197 bool Treaty::in_effect(void) const {
198  bool on_effect = false;
199  for(const auto& clause : this->clauses) {
200  if(clause->type == TreatyClauseType::WAR_REPARATIONS) {
201  const auto* dyn_clause = static_cast<const TreatyClause::WarReparations*>(clause);
202  on_effect = dyn_clause->in_effect();
203  } else if(clause->type == TreatyClauseType::ANEXX_PROVINCES) {
204  const auto* dyn_clause = static_cast<const TreatyClause::AnexxProvince*>(clause);
205  on_effect = dyn_clause->in_effect();
206  } else if(clause->type == TreatyClauseType::LIBERATE_NATION) {
207  const auto* dyn_clause = static_cast<const TreatyClause::LiberateNation*>(clause);
208  on_effect = dyn_clause->in_effect();
209  } else if(clause->type == TreatyClauseType::HUMILIATE) {
210  const auto* dyn_clause = static_cast<const TreatyClause::Humiliate*>(clause);
211  on_effect = dyn_clause->in_effect();
212  } else if(clause->type == TreatyClauseType::IMPOSE_POLICIES) {
213  const auto* dyn_clause = static_cast<const TreatyClause::ImposePolicies*>(clause);
214  on_effect = dyn_clause->in_effect();
215  } else if(clause->type == TreatyClauseType::CEASEFIRE) {
216  const auto* dyn_clause = static_cast<const TreatyClause::Ceasefire*>(clause);
217  on_effect = dyn_clause->in_effect();
218  }
219 
220  if(on_effect) {
221  break;
222  }
223  }
224  return on_effect;
225 }
