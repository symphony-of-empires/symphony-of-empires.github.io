36 // Uses the Aiias method to generate a lookup table of with different probabilties
37 template<typename T>
38 class DiscreteDistribution {
39 private:
40  std::vector<T> _items;
41  std::vector<float> alias;
42  std::vector<float> prob;
43 public:
44  DiscreteDistribution(std::vector<T> items, std::vector<float> probabilities)
45  : _items{ items }
46  {
47  // Scale each probabilty
48  float total = std::accumulate(probabilities.begin(), probabilities.end(), 0);
49  float scale = probabilities.size() / total;
50  for(size_t i = 0; i < probabilities.size(); i++) {
51  probabilities[i] *= scale;
52  }
53 
54  // Fill two work tables with probabilities larger/smaller than 1
55  std::vector<std::pair<float, size_t>> small;
56  std::vector<std::pair<float, size_t>> big;
57  for(size_t i = 0; i < probabilities.size(); i++) {
58  if(probabilities[i] < 1.0f) {
59  small.push_back(std::make_pair(probabilities[i], i));
60  }
61  else {
62  big.push_back(std::make_pair(probabilities[i], i));
63  }
64  }
65 
66  // Remove from the bigger one and place on the smaller ones
67  alias.resize(probabilities.size(), 0);
68  prob.resize(probabilities.size(), 0);
69  while(small.size() > 0 && big.size() > 0) {
70  std::pair<float, size_t> less = small.back();
71  small.pop_back();
72  std::pair<float, size_t> greater = big.back();
73  big.pop_back();
74  prob[less.second] = less.first;
75  alias[less.second] = greater.second;
76  greater.first = (greater.first + less.first) - 1;
77  if(greater.first < 1.f) {
78  small.push_back(greater);
79  } else {
80  big.push_back(greater);
81  }
82  }
83 
84  while(big.size() > 0) {
85  std::pair<float, int> greater = big.back();
86  big.pop_back();
87  prob[greater.second] = 1.f;
88  }
89 
90  while(small.size() > 0) {
91  std::pair<float, int> less = small.back();
92  small.pop_back();
93  prob[less.second] = 1.f;
94  }
95  }
96  ~DiscreteDistribution() {};
97 
98  // Get a random item with a certian probabilty
99  T get_item(void) {
100  size_t index = std::rand() % _items.size();
101  float r = static_cast <float> (rand()) / static_cast <float> (RAND_MAX);
102  if(prob[index] < r) {
103  index = alias[index];
104  }
105  return _items[index];
106  }
107 };
