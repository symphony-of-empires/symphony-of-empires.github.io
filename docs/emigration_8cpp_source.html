72 float nation_attraction(Nation* nation, Culture* culture) {
73  float attraction = nation->get_immigration_attraction_mod();
74  if(nation->is_accepted_culture(*culture)) {
75  // Linearized version, instead of using if-else trees we just
76  // multiply the attractive by the scale; EXTERMINATE = 3, so 3 - 3 is 0 which nullifies the attractivenes
77  // and the more open the borders are the more lenient the "scale" becomes
78  const int scale = 3 - nation->current_policy.treatment;
79  attraction *= scale;
80  }
81  return attraction;
82 }

89 void external_migration(World& world) {
90  std::vector<DiscreteDistribution<Province*>*> province_distributions;
91  for(auto nation : world.nations) {
92  std::vector<float> attractions;
93  std::vector<Province*> viable_provinces;
94  for(auto province : nation->owned_provinces) {
95  float attraction = province_attraction(province);
96  if(attraction <= 0) {
97  continue;
98  }
99 
100  attractions.push_back(attraction);
101  viable_provinces.push_back(province);
102  }
103 
104  DiscreteDistribution<Province*>* distribution = nullptr;
105  if(viable_provinces.size() > 0) {
106  distribution = new DiscreteDistribution(viable_provinces, attractions);
107  }
108  province_distributions.push_back(distribution);
109  }
110 
111  std::vector<DiscreteDistribution<Nation*>*> nation_distributions;
112  for(auto culture : world.cultures) {
113  std::vector<float> attractions;
114  std::vector<Nation*> viable_nations;
115  for(auto nation : world.nations) {
116  if(nation->owned_provinces.size() == 0) {
117  continue;
118  }
119 
120  if(nation->current_policy.migration == ALLOW_NOBODY) {
121  continue;
122  } else if(nation->current_policy.migration == ALLOW_ACCEPTED_CULTURES) {
123  if(!nation->accepted_cultures.count(culture)) {
124  continue;
125  }
126  }
127 
128  float attraction = nation_attraction(nation, culture);
129  if(attraction <= 0) {
130  continue;
131  }
132 
133  attractions.push_back(attraction);
134  viable_nations.push_back(nation);
135  }
136  DiscreteDistribution<Nation*>* distribution = nullptr;
137  if(viable_nations.size() > 0) {
138  distribution = new DiscreteDistribution(viable_nations, attractions);
139  }
140  nation_distributions.push_back(distribution);
141  }
142 
143  std::vector<Emigrated> emigration = std::vector<Emigrated>();
144  std::for_each(world.provinces.begin(), world.provinces.end(), [&province_distributions, &nation_distributions, &emigration, &world](auto& province) {
145  if(province->controller == nullptr) {
146  return;
147  }
148 
149  if(province->terrain_type->is_water_body) {
150  return;
151  }
152 
153  // Randomness factor to emulate a pseudo-imperfect economy
154  const float fuzz = std::fmod((float)(std::rand() + 1) / 1000.f, 2.f) + 1.f;
155 
156  for(size_t i = 0; i < province->pops.size(); i++) {
157  Pop& pop = province->pops[i];
158 
159  // Depending on how much not our life needs are being met is how many we
160  // want to get out of here
161  // And literacy determines "best" spot, for example a low literacy will
162  // choose a slightly less desirable location
163  const float emigration_willing = std::max<float>(-pop.life_needs_met * std::fmod(fuzz, 10), 0);
164  const ssize_t emigreers = std::fmod(pop.size * (emigration_willing / (std::rand() % 8)), pop.size) / 100;
165  if(emigreers > 0) {
166  // Check that laws on the province we are in allows for emigration
167  if(province->controller->current_policy.migration == ALLOW_NOBODY) {
168  continue;
169  }
170 
171  auto nation_distribution = nation_distributions[pop.culture->cached_id];
172  if(nation_distribution == nullptr) {
173  continue;
174  }
175 
176  auto nation = nation_distribution->get_item();
177  auto province_distribution = province_distributions[nation->cached_id];
178  if(province_distribution == nullptr) {
179  continue;
180  }
181 
182  auto choosen_province = province_distribution->get_item();
183  Emigrated emigrated = Emigrated(province->pops[i]);
184  emigrated.target = choosen_province;
185  emigrated.size = emigreers;
186  emigrated.origin = province;
187 
188  emigration.push_back(emigrated);
189  pop.size -= emigreers;
190  }
191  }
192  });
193 
194  // Now time to do the emigration - we will create a new POP on the province
195  // if a POP with similar culture, religion and type does not exist - and we
196  // will also subtract the amount of emigrated from the original POP to not
197  // create clones
198  for(const auto& target : emigration) {
199  auto new_pop = std::find(target.target->pops.begin(), target.target->pops.end(), target.emigred);
200  if(new_pop == target.target->pops.end()) {
201  Pop i_pop( target.emigred );
202  i_pop.size = target.size;
203  target.target->pops.push_back(i_pop);
204  } else {
205  new_pop->size += target.size;
206  new_pop->budget += target.emigred.budget;
207  }
208  }
209  emigration.clear();
210 }
