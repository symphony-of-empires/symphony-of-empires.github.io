38 // The purpouse of the serializer is to serialize objects onto a byte stream
39 // that can be transfered onto the disk or over the network.

96 // A serializer optimized to memcpy directly the element into the byte stream
97 // use only when the object can be copied without modification (i.e a class full of ints)
98 // The elements must have a fixed size for this to work.
99 template<typename T>
100 class SerializerMemcpy {
101 public:
102  static inline void serialize(Archive& ar, const T* obj) {
103  ar.expand(sizeof(T));
104  ar.copy_from(obj, sizeof(T));
105  }
106  static inline void deserialize(Archive& ar, T* obj) {
107  ar.copy_to(obj, sizeof(T));
108  }
109 };

147 // A serializer specialized in strings
148 // The serializer stores the lenght of the string and the string itself
149 // this is done so no errors can happen due to null stuff. (UTF-8 especially)
150 template<>
151 class Serializer<std::string> {
152 public:
153  static inline void serialize(Archive& ar, const std::string* obj) {
154  uint16_t len = obj->length();
155 
156  // Truncate lenght
157  if (len >= 1024)
158  len = 1024;
159 
160  // Put length for later deserialization (since UTF-8/UTF-16 exists)
161  ::serialize(ar, &len);
162 
163  // Copy the string into the output
164  if(len) {
165  ar.expand(len);
166  ar.copy_from(obj->c_str(), len);
167  }
168  }
169  static inline void deserialize(Archive& ar, std::string* obj) {
170  uint16_t len;
171 
172  // Obtain the lenght of the string to be read
173  ::deserialize(ar, &len);
174  if (len >= 1024)
175  throw SerializerException("String is too lenghty");
176 
177  // Obtain the string itself
178  char* string = new char[len + 1];
179 
180  if(len) {
181  ar.copy_to(string, len);
182  }
183  string[len] = '\0';
184 
185  *obj = string;
186  delete[] string;
187  }
188 };

190 // Non-contigous serializer for STL containers
191 // This serializer class works primarly with containers whose memory is contiguous
192 template<typename T, typename C>
193 class SerializerContainer {
194 public:
195  static inline void serialize(Archive& ar, const C* obj_group) {
196  uint32_t len = obj_group->size();
197  ::serialize(ar, &len);
198  for(auto& obj: *obj_group) {
199  Serializer<T>::serialize(ar, &obj);
200  }
201  }
202  static inline void deserialize(Archive& ar, C* obj_group) {
203  uint32_t len;
204  ::deserialize(ar, &len);
205  obj_group->clear();
206  for(size_t i = 0; i < len; i++) {
207  T obj;
208  Serializer<T>::deserialize(ar, &obj);
209  obj_group->insert(obj);
210  }
211  }
212 };

287 // Used as a template for serializable objects (pointers mostly) which should be
288 // treated as a reference instead of the object itself
289 template<typename W, typename T>
290 class SerializerReference {
291 public:
292  static inline void serialize(Archive& stream, const T* const* obj) {
293  typename T::Id id = (*obj == nullptr) ? (typename T::Id)-1 : W::get_instance().get_id(*obj);
294  ::serialize(stream, &id);
295  };
296 
297  static inline void deserialize(Archive& stream, T** obj) {
298  typename T::Id id;
299  ::deserialize(stream, &id);
300  if(id >= W::get_instance().get_list(*obj).size()) {
301  *obj = nullptr;
302  return;
303  }
304  *obj = (id != (typename T::Id)-1) ? W::get_instance().get_list(*obj)[id] : nullptr;
305  };
306 };
