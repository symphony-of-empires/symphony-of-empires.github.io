35  class ThreadPool {
36  // An atomic all idling threads use
37  std::atomic<bool> running;
38  // While std::thread is not copy-able, the std::vector STL container
39  // allows for movable elements instead. keep this in mind when a bug happens
40  std::vector<std::thread> threads;
41  // Queues are suited for the job here, since first-requested-jobs
42  // should have higher priority than the last one, this allows sequential
43  // completion of tasks
44  std::queue<std::function<void()>> jobs;
45  std::mutex job_mutex; // Our mutex for preventing race conditions on the queue above
46  std::condition_variable cv_task; // Called when job is added
47  std::condition_variable cv_finished; // Called when job is finished
48  size_t busy = 0; // Amount of threads currently working

51  ThreadPool() {
52  // It's a good idea to use as many threads as the hardware implementation
53  // supports. Otherwise we can run into performance hits.

55  this->running = true;

67  ~ThreadPool() {
68  // We are going to signal all threads to shutdown
69  std::unique_lock<std::mutex> latch(job_mutex);
70  running = false;
71  cv_task.notify_all();
72  latch.unlock();
73 
74  // After that we will start joining all threads - if a thread is executing
75  // by a prolonged time, it will block the entire process
76  for(auto job = threads.begin(); job != threads.end(); job++)

90  void thread_loop() {
91  while(true) {
92  std::unique_lock<std::mutex> latch(job_mutex);
93  cv_task.wait(latch, [this]() {
94  return (!running || !jobs.empty());
95  });
96 
97  if(!running) break;
98 
99  // pull from queue
100  std::function<void()> fn = jobs.front();
101  jobs.pop();
102 
103  busy++;
104  // release lock. run async
105  latch.unlock();
106 
107  fn();
108 
109  latch.lock();
110  cv_finished.notify_one();
111  }
112  }

118  return (jobs.empty() && (busy == 0));
