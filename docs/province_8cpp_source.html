60 float Province::get_attractiveness(const Pop& pop) const {
61  float attractive = this->base_attractive;
62 
63  if(!this->owner->is_accepted_culture(pop) && !this->owner->is_accepted_religion(pop)) {
64  // Linearized version, instead of using if-else trees we just
65  // multiply the attractive by the scale; EXTERMINATE = 3, so 3 - 3 is 0 which nullifies the attractivenes
66  // and the more open the borders are the more lenient the "scale" becomes
67  const int scale = 3 - this->owner->current_policy.treatment;
68  attractive *= scale;
69  }
70 
71  // Account for literacy difference
72  attractive -= this->owner->base_literacy;
73 
74  // Account for GDP difference
75  attractive -= this->owner->gdp * 0.0001f;
76 
77  // A social value between 0 and 1 is for poor people, the value for medium class
78  // is between 1 and 2, for the rich is above 2
79 
80  // For the lower class, lower taxes is good, and so on for other POPs
81  if(pop.type->social_value >= 0.f && pop.type->social_value <= 1.f) {
82  attractive += -(this->owner->current_policy.poor_flat_tax);
83  }
84  // For the medium class
85  else if(pop.type->social_value >= 1.f && pop.type->social_value <= 2.f) {
86  attractive += -(this->owner->current_policy.med_flat_tax);
87  }
88  // For the high class
89  else if(pop.type->social_value >= 2.f) {
90  attractive += -(this->owner->current_policy.rich_flat_tax);
91  }
92  return attractive;
93 }
