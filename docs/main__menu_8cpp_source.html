49 MainMenuConnectServer::MainMenuConnectServer(GameState& _gs)
50  : UI::Window(0, 0, 512, 128, nullptr),
51  gs{ _gs }
52 {
53  this->is_scroll = false;
54  this->text("Internet multiplayer");
55 
56  ip_addr_inp = new UI::Input(0, 0, 128, 24, this);
57  ip_addr_inp->buffer = "127.0.0.1";
58  ip_addr_inp->text(ip_addr_inp->buffer);
59  ip_addr_inp->tooltip = new UI::Tooltip(ip_addr_inp, 512, 24);
60  ip_addr_inp->tooltip->text("IP Address of the server");
61 
62  username_inp = new UI::Input(0, 24, 512, 24, this);
63  username_inp->buffer = "Player";
64  username_inp->text(username_inp->buffer);
65  username_inp->tooltip = new UI::Tooltip(username_inp, 512, 24);
66  username_inp->tooltip->text("Your publicly visible username");
67 
68  conn_btn = new UI::Button(0, 48, 128, 24, this);
69  conn_btn->user_data = this;
70  conn_btn->text("Connect");
71  conn_btn->on_click = ([](UI::Widget& w, void*) {
72  auto& o = static_cast<MainMenuConnectServer&>(*w.parent);
73  print_info("Okey, connecting to [%s]", o.ip_addr_inp->buffer.c_str());
74 
75  // TODO: Handle when mods differ (i.e checksum not equal to host)
76  o.gs.host_mode = false;
77 
78  try {
79  o.gs.client = new Client(o.gs, o.ip_addr_inp->buffer, 1836);
80  o.gs.client->username = o.username_inp->buffer;
81  o.gs.client->wait_for_snapshot();
82  } catch(UnifiedRender::Networking::SocketException& e) {
83  o.gs.ui_ctx->prompt("Network layer error", e.what());
84  goto failure_cleanup;
85  } catch(ClientException& e) {
86  o.gs.ui_ctx->prompt("Client error", e.what());
87  goto failure_cleanup;
88  } catch(ServerException& e) {
89  o.gs.ui_ctx->prompt("Server error", e.what());
90  goto failure_cleanup;
91  }
92  o.gs.in_game = true;
93  failure_cleanup:
94  delete o.gs.world;
95  delete o.gs.client;
96  });
97 
98  auto* close_btn = new UI::CloseButton(0, 0, 128, 24, this);
99  close_btn->below_of(*conn_btn);
100  close_btn->text("Cancel");
101 }
