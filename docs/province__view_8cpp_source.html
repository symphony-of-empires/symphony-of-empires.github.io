45 void ProvincePopulationTab::update_piecharts() {
46  // Obtain population information
47  std::vector<size_t> culture_sizes(gs.world->cultures.size(), 0);
48  std::vector<size_t> religion_sizes(gs.world->religions.size(), 0);
49  std::vector<size_t> pop_type_sizes(gs.world->pop_types.size(), 0);
50  for(const auto& pop : province->pops) {
51  culture_sizes[gs.world->get_id(pop.culture)] += pop.size;
52  religion_sizes[gs.world->get_id(pop.religion)] += pop.size;
53  pop_type_sizes[gs.world->get_id(pop.type)] += pop.size;
54  }
55 
56  std::vector<UI::ChartData> cultures_data, religions_data, pop_types_data;
57  for(const auto& culture : gs.world->cultures) {
58  cultures_data.push_back(UI::ChartData(culture_sizes[gs.world->get_id(culture)], culture->name, UnifiedRender::Color::rgba32(culture->color)));
59  }
60  cultures_pie->set_data(cultures_data);
61 
62  for(const auto& religion : gs.world->religions) {
63  religions_data.push_back(UI::ChartData(religion_sizes[gs.world->get_id(religion)], religion->name, UnifiedRender::Color::rgba32(religion->color)));
64  }
65  religions_pie->set_data(religions_data);
66 
67  for(const auto& pop_type : gs.world->pop_types) {
68  const auto color = UnifiedRender::Color(
69  (uint8_t)((gs.world->get_id(pop_type) * 12) % 256),
70  (uint8_t)((gs.world->get_id(pop_type) * 31) % 256),
71  (uint8_t)((gs.world->get_id(pop_type) * 97) % 256)
72  );
73  pop_types_data.push_back(UI::ChartData(pop_type_sizes[gs.world->get_id(pop_type)], pop_type->name, color));
74  }
75  pop_types_pie->set_data(pop_types_data);
76 
77  if(pop_infos.size() < province->pops.size()) {
78  for(size_t i = pop_infos.size(); i < province->pops.size(); i++) {
79  PopInfo* info = new PopInfo(gs, 0, (i * 24) + 128, province, i, this);
80  pop_infos.push_back(info);
81  }
82  } else if(pop_infos.size() > province->pops.size()) {
83  for(size_t i = province->pops.size(); i < pop_infos.size(); i++) {
84  pop_infos[i]->kill();
85  }
86  pop_infos.erase(pop_infos.begin() + province->pops.size(), pop_infos.end());
87  }
88 }

144 
145  products_data.push_back(UI::ChartData(product->demand, product->good->name, product_col));
146  }
147  o.products_pie->set_data(products_data);
148  });
149 
150  // Initial product info

152  for(const auto& product : this->province->products) {

154  continue;

158  this->product_infos.push_back(info);
