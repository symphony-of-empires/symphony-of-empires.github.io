100 void Tooltip::text(const std::string& text) {
101  // Delete old labels in vector (if any)
102  for(auto& lab : labels) {
103  lab->kill();
104  }
105  labels.clear();
106 
107  if(text.empty()) {
108  return;
109  }
110 
111  // Separate the text in multiple labels and break on space
112  // TODO: only works for monospace fonts width width 12, fix it for all fonts
113  size_t pos = 0, y = 0;
114  size_t line_width = std::max<size_t>(1, this->width / 12);
115  while(pos < text.length()) {
116  size_t remaining_chars = text.length() - pos;
117  size_t end_pos = text.length();
118  if(remaining_chars > line_width) {
119  end_pos = pos + line_width;
120  }
121  bool break_line = false;
122  for(int i = pos; i <= end_pos; i++) {
123  if(text[i] == '\n') {
124  end_pos = i;
125  break_line = true;
126  break;
127  }
128  }
129  if(!break_line && remaining_chars > line_width) {
130  for(int i = end_pos; i > pos; i--) {
131  if(text[i] == ' ') {
132  end_pos = i;
133  break;
134  }
135  }
136  }
137 
138  std::string buf = text.substr(pos, end_pos - pos);
139  pos = end_pos;
140  if (break_line) {
141  pos++;
142  }
143 
144  UI::Label* lab = new UI::Label(8, y, buf, this);
145  labels.push_back(lab);
146 
147  y += 24;
148  }
149  height = y;
150 }
