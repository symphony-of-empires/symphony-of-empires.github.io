52 void Widget::draw_rect(const GLuint tex,
53  UnifiedRender::Rect rect_pos,
54  UnifiedRender::Rect rect_tex,
55  UnifiedRender::Rect viewport) {
56 
57  glm::vec2 pos_size = rect_pos.size();
58  pos_size.x = pos_size.x > 0? pos_size.x : 1.f;
59  pos_size.y = pos_size.y > 0? pos_size.y : 1.f;
60  glm::vec2 tex_size = rect_tex.size();
61 
62  if(rect_pos.left < viewport.left) {
63  float x_ratio = (viewport.left - rect_pos.left) / pos_size.x;
64  rect_tex.left = rect_tex.left + x_ratio * tex_size.x;
65  rect_pos.left = viewport.left;
66  }
67  if(rect_pos.right > viewport.right) {
68  float x_ratio = (rect_pos.right - viewport.right) / pos_size.x;
69  rect_tex.right = rect_tex.right - x_ratio * tex_size.x;
70  rect_pos.right = viewport.right;
71  }
72  if(rect_pos.top < viewport.top) {
73  float y_ratio = (viewport.top - rect_pos.top) / pos_size.y;
74  rect_tex.top = rect_tex.top + y_ratio * tex_size.y;
75  rect_pos.top = viewport.top;
76  }
77  if(rect_pos.bottom > viewport.bottom) {
78  float y_ratio = (rect_pos.bottom - viewport.bottom) / pos_size.y;
79  rect_tex.bottom = rect_tex.bottom - y_ratio * tex_size.y;
80  rect_pos.bottom = viewport.bottom;
81  }
82 
83  glBindTexture(GL_TEXTURE_2D, tex);
84  glBegin(GL_TRIANGLES);
85  glTexCoord2f(rect_tex.left, rect_tex.top);
86  glVertex2f(rect_pos.left, rect_pos.top);
87  glTexCoord2f(rect_tex.right, rect_tex.top);
88  glVertex2f(rect_pos.right, rect_pos.top);
89  glTexCoord2f(rect_tex.right, rect_tex.bottom);
90  glVertex2f(rect_pos.right, rect_pos.bottom);
91 
92  glTexCoord2f(rect_tex.right, rect_tex.bottom);
93  glVertex2f(rect_pos.right, rect_pos.bottom);
94  glTexCoord2f(rect_tex.left, rect_tex.bottom);
95  glVertex2f(rect_pos.left, rect_pos.bottom);
96  glTexCoord2f(rect_tex.left, rect_tex.top);
97  glVertex2f(rect_pos.left, rect_pos.top);
98  glEnd();
99 }

101 void Widget::draw_border(Border* border, UnifiedRender::Rect viewport) {
102  float x_offset = border->offset.x;
103  float y_offset = border->offset.y;
104  float b_w = border->size.x;
105  float b_h = border->size.y;
106  float b_tex_w = border->texture_size.x;
107  float b_tex_h = border->texture_size.y;
108  auto border_tex = border->texture;
109 
110  // Draw border edges and corners
111  UnifiedRender::Rect pos_rect(0, 0, 0, 0);
112  UnifiedRender::Rect tex_rect(0, 0, 0, 0);
113  pos_rect.left = x_offset;
114  pos_rect.top = y_offset;
115  pos_rect.right = x_offset + b_w;
116  pos_rect.bottom = y_offset + b_h;
117  tex_rect.left = 0;
118  tex_rect.top = 0;
119  tex_rect.right = b_tex_w / border_tex->width;
120  tex_rect.bottom = b_tex_h / border_tex->height;

124  draw_rect(tex, pos_rect, tex_rect, viewport);
125

129  pos_rect.right = width;
130  tex_rect.right = 1.f;
131  draw_rect(tex, pos_rect, tex_rect, viewport);
132 
133  // Bottom right corner

135  tex_rect.top = (border_tex->height - b_tex_h) / border_tex->height;
136  pos_rect.bottom = height;

139 
140  // Bottom left corner

143  pos_rect.right = x_offset + b_w;

326 void Widget::recalc_child_pos() {
327  if(flex == Flex::NONE)
328  return;
329 
330  bool is_row = flex == Flex::ROW;
331  size_t lenght = 0;
332  for(auto& child : children) {
333  lenght += is_row ? child->width : child->height;
334  }
335 
336  size_t current_lenght = 0;
337  int size = 0, difference = 0;
338  switch(flex_justify)
339  {
340  case FlexJustify::START:
341  current_lenght = 0;
342  for(auto& child : children) {
343  if(is_row) {
344  child->x = current_lenght;
345  current_lenght += child->width + flex_gap;
346  }
347  else {
348  child->y = current_lenght;
349  current_lenght += child->height + flex_gap;
350  }
351  }
352  break;
353  case FlexJustify::END:
354  current_lenght = is_row ? width : height;
355  for(auto& child : children) {
356  if(is_row) {
357  child->x = current_lenght - child->width - flex_gap;
358  current_lenght -= child->width;
359  }
360  else {
361  child->y = current_lenght - child->height - flex_gap;
362  current_lenght -= child->height;
363  }
364  }
365  break;
366  case FlexJustify::SPACE_BETWEEN:
367  current_lenght = 0;
368  size = is_row ? width : height;
369  difference = (size - lenght) / children.size();
370  for(auto& child : children) {
371  if(is_row) {
372  child->x = current_lenght;
373  current_lenght += child->width + difference;
374  }
375  else {
376  child->y = current_lenght;
377  current_lenght += child->height + difference;
378  }
379  }
380  break;
381  case FlexJustify::SPACE_AROUND:
382  size = is_row ? width : height;
383  difference = (size - lenght) / (children.size() + 1);
384  current_lenght = std::max<int>(0, difference);
385  for(auto& child : children) {
386  if(is_row) {
387  child->x = current_lenght;
388  current_lenght += child->width + difference;
389  }
390  else {
391  child->y = current_lenght;
392  current_lenght += child->height + difference;
393  }
394  }
395  break;
396  }
397  switch(flex_align)
398  {
399  case FlexAlign::START:
400  for(auto& child : children) {
401  if(is_row) {
402  child->y = 0;
403  }
404  else {
405  child->x = 0;
406  }
407  }
408  break;
409  case FlexAlign::END:
410  for(auto& child : children) {
411  if(is_row) {
412  child->y = std::max<int>(0, height - child->height);
413  }
414  else {
415  child->x = std::max<int>(0, width - child->width);
416  }
417  }
418  break;
419  case FlexAlign::CENTER:
420  for(auto& child : children) {
421  if(is_row) {
422  child->y = std::max<int>(0, height - child->height) / 2;
423  }
424  else {
425  child->x = std::max<int>(0, width - child->width) / 2;
426  }
427  }
428  break;
429  }
430 }

444 static unsigned int power_two_floor(unsigned int val) {
445  unsigned int power = 2, nextVal = power * 2;
446 
447  while((nextVal *= 2) <= val) {
448  power*=2;
449  }
450 
451  return power*2;

453 void Widget::text(const std::string& _text) {
454  // Copy _text to a local scope (SDL2 does not like references)
455  if(text_texture != nullptr) {
456  // Auto deletes gl_texture
457  delete text_texture;
458  text_texture = nullptr;
459  }
460 
461  if(_text.empty()) {
462  return;
463  }
464 
465  //TTF_SetFontStyle(g_ui_context->default_font, TTF_STYLE_BOLD);
466  SDL_Color black_color = { 0, 0, 0, 0 };
467  SDL_Surface* surface = TTF_RenderUTF8_Blended(g_ui_context->default_font, _text.c_str(), black_color);
468  if(surface == nullptr) {
469  throw std::runtime_error(std::string() + "Cannot create text surface: " + TTF_GetError());
470  }
471 
472  int w = power_two_floor(surface->w) * 2;
473  int h = power_two_floor(surface->h) * 2;
474  //Create a surface to the correct size in RGB format, and copy the old image
475  SDL_Surface* s = SDL_CreateRGBSurface(0, w, h, 32, 0x00ff0000,0x0000ff00,0x000000ff,0xff000000);
476  SDL_BlitSurface(surface, NULL, s, NULL);
477  SDL_FreeSurface(surface);
478  surface = s;
479 
480  text_texture = new UnifiedRender::Texture(surface->w, surface->h);
481  text_texture->gl_tex_num = 0;
482  text_texture->to_opengl(surface);
483  SDL_FreeSurface(surface);
484 
485  const char* error_msg = SDL_GetError();
486  if(error_msg[0] != '\0') {
487  print_error("SDL error %s", error_msg);
488  }
489 }
