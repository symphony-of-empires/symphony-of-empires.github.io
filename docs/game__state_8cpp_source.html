124 void handle_event(Input& input, GameState& gs) {
125  std::pair<int, int>& mouse_pos = input.mouse_pos;
126  UI::Context* ui_ctx = gs.ui_ctx;
127 
128  int& width = gs.width;
129  int& height = gs.height;
130  SDL_Event event;
131 
132  // Check window size every update
133  // - needed cause the window sometimes changes size without calling the change window size event
134  SDL_GetWindowSize(gs.window, &width, &height);
135  gs.ui_ctx->resize(width, height);
136  gs.map->camera->set_screen(width, height);
137 
138  while(SDL_PollEvent(&event)) {
139  bool click_on_ui = false;
140  switch(event.type) {
141  case SDL_CONTROLLERDEVICEADDED:
142  break;
143  case SDL_MOUSEBUTTONDOWN:
144  SDL_GetMouseState(&mouse_pos.first, &mouse_pos.second);
145  click_on_ui = ui_ctx->check_hover(mouse_pos.first, mouse_pos.second);
146  ui_ctx->check_drag(mouse_pos.first, mouse_pos.second);
147  if(event.button.button == SDL_BUTTON_MIDDLE) {
148  input.middle_mouse_down = true;
149  }
150  break;
151  case SDL_JOYBUTTONDOWN:
152  ui_ctx->check_drag(mouse_pos.first, mouse_pos.second);
153  break;
154  case SDL_MOUSEBUTTONUP:
155  SDL_GetMouseState(&mouse_pos.first, &mouse_pos.second);
156  if(event.button.button == SDL_BUTTON_MIDDLE) {
157  input.middle_mouse_down = false;
158  break;
159  }
160 
161  click_on_ui = ui_ctx->check_click(mouse_pos.first, mouse_pos.second);
162  if(!click_on_ui && gs.current_mode != MapMode::NO_MAP) {
163  gs.map->handle_click(gs, event);
164  }
165 
166  if(click_on_ui) {
167  std::scoped_lock lock(gs.sound_lock);
168  auto entries = Path::get_all_recursive("sfx/click");
169  if(!entries.empty()) {
170  gs.sound_queue.push_back(new UnifiedRender::Audio(entries[std::rand() % entries.size()]));
171  }
172  }
173  break;
174  case SDL_JOYBUTTONUP:
175  click_on_ui = ui_ctx->check_click(mouse_pos.first, mouse_pos.second);
176  if(!click_on_ui && gs.current_mode != MapMode::NO_MAP) {
177  gs.map->handle_click(gs, event);
178  }
179 
180  if(click_on_ui) {
181  std::scoped_lock lock(gs.sound_lock);
182  auto entries = Path::get_all_recursive("sfx/click");
183  if(!entries.empty()) {
184  gs.sound_queue.push_back(new UnifiedRender::Audio(entries[std::rand() % entries.size()]));
185  }
186  }
187  break;
188  case SDL_MOUSEMOTION:
189  SDL_GetMouseState(&mouse_pos.first, &mouse_pos.second);
190  click_on_ui = ui_ctx->check_hover(mouse_pos.first, mouse_pos.second);
191  break;
192  case SDL_MOUSEWHEEL:
193  SDL_GetMouseState(&mouse_pos.first, &mouse_pos.second);
194  ui_ctx->check_hover(mouse_pos.first, mouse_pos.second);
195  click_on_ui = ui_ctx->check_wheel(mouse_pos.first, mouse_pos.second, event.wheel.y * 6);
196  break;
197  case SDL_TEXTINPUT:
198  ui_ctx->check_text_input((const char*)&event.text.text);
199  break;
200  case SDL_KEYDOWN:
201  switch(UnifiedRender::Keyboard::from_sdlk(event.key.keysym.sym)) {
202  case UnifiedRender::Keyboard::Key::F1:
203  if(gs.current_mode == MapMode::NORMAL) {
204  if(gs.profiler_view) {
205  delete gs.profiler_view;
206  gs.profiler_view = nullptr;
207  } else {
208  gs.profiler_view = new Interface::ProfilerView(gs);
209  }
210  }
211  break;
212  case UnifiedRender::Keyboard::Key::SPACE:
213  if(gs.editor) {
214  break;
215  }
216 
217  if(gs.current_mode == MapMode::NORMAL) {
218  gs.paused = !gs.paused;
219  if(gs.paused) {
220  ui_ctx->prompt("Control", "Unpaused");
221  }
222  else {
223  ui_ctx->prompt("Control", "Paused");
224  }
225  }
226  break;
227  case UnifiedRender::Keyboard::Key::B:
228  if(gs.editor) {
229  break;
230  }
231 
232  if(gs.current_mode == MapMode::NORMAL) {
233  if(input.select_pos.first < gs.world->width || input.select_pos.second < gs.world->height) {
234  const Tile& tile = gs.world->get_tile(input.select_pos.first, input.select_pos.second);
235  if(tile.province_id >= gs.world->provinces.size()) {
236  break;
237  }
238 
239  new Interface::BuildingBuildView(gs, input.select_pos.first, input.select_pos.second, true, gs.world->provinces[tile.province_id]->owner, gs.world->provinces[tile.province_id]);
240  }
241  }
242  break;
243  case UnifiedRender::Keyboard::Key::BACKSPACE:
244  ui_ctx->check_text_input(nullptr);
245  break;
246  }
247  break;
248  case SDL_JOYAXISMOTION:
249  ui_ctx->check_hover(gs.input.mouse_pos.first, gs.input.mouse_pos.second);
250  break;
251  case SDL_QUIT:
252  gs.run = false;
253  gs.paused = false;
254  break;
255  default:
256  break;
257  }
258 
259  if(gs.current_mode != MapMode::NO_MAP && !click_on_ui) {
260  gs.map->update(event, input, ui_ctx);
261  }
262  }
263  ui_ctx->clear_dead();
264 }

274 void handle_popups(std::vector<Event*>& displayed_events, std::vector<Treaty*>& displayed_treaties, GameState& gs) {
275  // Put popups
276  // Event + Descision popups
277  for(auto& msg : gs.curr_nation->inbox) {
278  // Check that the event is not already displayed to the user
279  auto iter = std::find_if(displayed_events.begin(), displayed_events.end(), [&msg](const auto& e) { return e->ref_name == msg->ref_name; });
280  if(iter != displayed_events.end()) {
281  continue;
282  }
283 
284  new Interface::DescisionWindow(gs, *msg);
285  displayed_events.push_back(msg);
286  }
287 
288  for(auto& treaty : gs.world->treaties) {
289  // Check that the treaty is not already displayed
290  auto iter = std::find_if(displayed_treaties.begin(), displayed_treaties.end(), [&treaty](const auto& e) { return e == treaty; });
291  if(iter != displayed_treaties.end()) {
292  continue;
293  }
294 
295  // Do not mess with treaties we don't partake in, hehe
296  if(!treaty->does_participate(*gs.curr_nation)) {
297  continue;
298  }
299 
300  // Must participate in treaty
301  new Interface::TreatyChooseWindow(gs, treaty);
302  displayed_treaties.push_back(treaty);
303  }
304 }

341 void main_loop(GameState& gs) {
342  gs.input = Input();
343  gs.in_game = false;
344 
345  // Connect to server prompt
346  gs.current_mode = MapMode::NO_MAP;
347  //auto* mm_bg = new UI::Image(0, 0, gs.width, gs.height, &UnifiedRender::State::get_instance().tex_man->load(Path::get("gfx/globe.png")));
348  //mm_bg->is_fullscreen = true;

355  std::vector<Event*> displayed_events;
356  std::vector<Treaty*> displayed_treaties;
357 
358  // Call update_on_tick on start of the gamestate
359  gs.update_tick = true;
360  gs.run = true;
361  gs.paused = true;
362 
363  // Start the world thread
364  std::thread world_th(&GameState::world_thread, &gs);
365  while(gs.run) {
366  handle_event(gs.input, gs);
367  if(gs.current_mode == MapMode::NORMAL) {
368  if(gs.world->world_mutex.try_lock()) {
369  handle_popups(displayed_events, displayed_treaties, gs);
370  gs.world->world_mutex.unlock();
371  }
372  }
373 
374  if(gs.sound_queue.empty()) {
375  for(const auto& war : gs.world->wars) {
376  for(const auto& battle : war->battles) {
377  std::scoped_lock lock(gs.sound_lock);
378  auto entries = Path::get_all_recursive("sfx/gunfire");
379  if(!entries.empty()) {
380  gs.sound_queue.push_back(new UnifiedRender::Audio(entries[std::rand() % entries.size()]));
381  }
382  break;
383  }
384  break;
385  }
386  }
387 
388  if(gs.update_tick) {
389  gs.update_on_tick();
390  gs.update_tick = false;
391 
392  if(gs.current_mode == MapMode::NORMAL) {
393  if(gs.world->world_mutex.try_lock()) {
394  // Production queue
395  if(!gs.production_queue.empty()) {
396  for(unsigned int i = 0; i < gs.production_queue.size(); i++) {
397  UnitType* unit = gs.production_queue[i];
398 
399  // TODO: Make a better queue AI
400  bool is_built = false;
401  for(const auto& building : gs.world->buildings) {
402  // Must be our building
403  if(building->get_owner() != gs.curr_nation) {
404  continue;
405  }
406 
407  // Must not be working on something else
408  if(building->working_unit_type != nullptr) {
409  continue;
410  }
411 
412  is_built = true;
413 
414  g_client->send(Action::BuildingStartProducingUnit::form_packet(building, unit));
415  break;
416  }
417 
418  if(!is_built) {
419  continue;
420  }
421 
422  gs.production_queue.erase(gs.production_queue.begin() + i);
423  i--;
424  }
425  }
426  gs.world->world_mutex.unlock();
427  }
428  }
429 
430  if(gs.music_queue.empty()) {
431  // Search through all the music in 'music/ambience' and picks a random
432  auto entries = Path::get_all_recursive("music/ambience");
433  if(entries.size() != 0) {
434  int music_index = std::rand() % entries.size();
435  std::scoped_lock lock(gs.sound_lock);
436  gs.music_fade_value = 100.f;
437  gs.music_queue.push_back(new UnifiedRender::Audio(entries[music_index]));
438  }
439  }
440  }
441 
442  std::scoped_lock lock(gs.render_lock);
443 
444  if(!gs.motion_blur) {
445  gs.clear();
446  }
447 
448  if(gs.current_mode != MapMode::NO_MAP) {
449  std::scoped_lock lock(gs.world->world_mutex);
450  gs.map->draw(gs);
451  gs.map->camera->update();
452  }
453  gs.ui_ctx->render_all();
454 
455  glUseProgram(0);
456  glActiveTexture(GL_TEXTURE0);
457  glViewport(0, 0, gs.width, gs.height);
458  glPushMatrix();
459  glMatrixMode(GL_PROJECTION);
460  glLoadIdentity();

462  glMatrixMode(GL_MODELVIEW);
463  glLoadIdentity();

466  glPushMatrix();

468  gs.tex_man->load(Path::get("gfx/cursor_b.png")).bind();
469  glColor3f(1.f, 1.f, 1.f);
470  glBegin(GL_TRIANGLES);
471  glTexCoord2f(0.f, 0.f);
472  glVertex2f(0.f, 0.f);
473  glTexCoord2f(1.f, 0.f);
474  glVertex2f(32.f, 0.f);
475  glTexCoord2f(1.f, 1.f);
476  glVertex2f(32.f, 32.f);

496 
497  if(1) {
498  FILE* fp = fopen(Path::get("locale/ro/main.po").c_str(), "rt");
499  if(fp != nullptr) {
500  char* tmp = new char[1000];
501  while(fgets(tmp, 1000, fp) != nullptr) {
502  if(!strncmp(tmp, "msgid", 5)) {
503  char* msgid = new char[100];
504  sscanf(tmp + 5, " %*c%[^\"]s%*c ", msgid);
505  fgets(tmp, 1000, fp);
506  if(!strncmp(tmp, "msgstr", 6)) {
507  char* msgstr = new char[100];
508  sscanf(tmp + 6, " %*c%[^\"]s%*c ", msgstr);
509  trans_msg[msgid] = std::string(msgstr);
510  delete[] msgstr;
511  }
512  delete[] msgid;
513  }
514  }
515  delete[] tmp;
516  fclose(fp);
517 
518  for(const auto& [key, value] : trans_msg) {
519  print_info("%s=%s", key.c_str(), value.c_str());
520  }
521  }
522  else {
523  print_error("Cannot open locale file %s", Path::get("locale/ko/main.po").c_str());
524  }
525  }
526 
527  gs.ui_ctx = new UI::Context();
528 
529  gs.ui_ctx->resize(gs.width, gs.height);
530 
531  gs.loaded_world = false;
532  gs.loaded_map = false;
533 
534  gs.world = new World();
535  gs.world->load_initial();
536  gs.world->load_mod();
537  gs.map = new Map(*gs.world, gs.width, gs.height);
538 
539  main_loop(gs);
540  return;
