60 // This is the handling thread-function for handling a connection to a single client
61 // Sending packets will only be received by the other end, when trying to broadcast please
62 // put the packets on the send queue, they will be sent accordingly
63 void Server::net_loop(int id) {
64  UnifiedRender::Networking::ServerClient& cl = clients[id];
65  while(run) {
66  int conn_fd = 0;
67  try {
68  cl.is_connected = false;
69  while(!cl.is_connected) {
70  conn_fd = cl.try_connect(fd);
71  if(!run) {
72  throw ServerException("Server closed");
73  }
74  }
75 
76  Nation* selected_nation = nullptr;
77  UnifiedRender::Networking::Packet packet = UnifiedRender::Networking::Packet(conn_fd);
78 
79  player_count++;
80 
81  // Send the whole snapshot of the world (only we are not the host)
82  // We also need to make sure the global client is not nullptr AND
83  // we also will only NOT send the snapshot to the first client only
84  /*if(!gs.host_mode && player_count <= 1) {
85  Archive ar = Archive();
86  std::scoped_lock lock(g_world->world_mutex);
87  ::serialize(ar, g_world);
88  packet.send(ar.get_buffer(), ar.size());
89  }*/
90 
91  // Read the data from client
92  {
93  Archive ar = Archive();
94  packet.recv();
95  ar.set_buffer(packet.data(), packet.size());
96 
97  ActionType action;
98  ::deserialize(ar, &action);
99  ::deserialize(ar, &cl.username);
100  }
101 
102  // Tell all other clients about the connection of this new client
103  {
104  Archive ar = Archive();
105  ActionType action = ActionType::CONNECT;
106  ::serialize(ar, &action);
107  packet.data(ar.get_buffer(), ar.size());
108  broadcast(packet);
109  }
110 
111  ActionType action = ActionType::PING;
112  packet.send(&action);
113 
114  // TODO: The world mutex is not properly unlocked when an exception occours
115  // this allows clients to abruptly disconnect and softlock a server
116  // so we did this little trick of manually unlocking - but this is a bad idea
117  // we need to use RAII to just unlock the thing on destruction *(specifically when
118  // an exception is thrown), since we are using C++
119  Archive ar = Archive();
120  while(run && cl.is_connected == true) {
121  cl.flush_packets();
122 
123  // Check if we need to read packets
124  if(cl.has_pending()) {
125  packet.recv();
126  ar.set_buffer(packet.data(), packet.size());
127  ar.rewind();
128  ::deserialize(ar, &action);
129 
130  print_info("Receiving %zu bytes from #%zu", packet.size(), (size_t)id);
131 
132  if(selected_nation == nullptr && (action != ActionType::PONG && action != ActionType::CHAT_MESSAGE && action != ActionType::SELECT_NATION))
133  throw ServerException("Unallowed operation without selected nation");
134 
135  switch(action) {
136  // - Used to test connections between server and client
137  case ActionType::PONG:
138  action = ActionType::PING;
139  packet.send(&action);
140  print_info("Received pong, responding with ping!");
141  break;
142  // - Client tells server to enact a new policy for it's nation
143  case ActionType::NATION_ENACT_POLICY: {
144  Policies policies;
145  ::deserialize(ar, &policies);
146 
147  // TODO: Do parliament checks and stuff
148  selected_nation->current_policy = policies;
149  } break;
150  // - Client tells server to change target of unit
151  case ActionType::UNIT_CHANGE_TARGET: {
152  Unit* unit;
153  ::deserialize(ar, &unit);
154  if(unit == nullptr)
155  throw ServerException("Unknown unit");
156 
157  // Must control unit
158  if(selected_nation != unit->owner)
159  throw ServerException("Nation does not control unit");
160 
161  Province* province;
162  ::deserialize(ar, &province);
163  if(province != nullptr)
164  print_info("Unit changes targets to %s", province->ref_name.c_str());
165 
166  unit->set_target(*province);
167  } break;
168  // Client tells the server about the construction of a new unit, note that this will
169  // only make the building submit "construction tickets" to obtain materials to build
170  // the unit can only be created by the server, not by the clients
171  case ActionType::BUILDING_START_BUILDING_UNIT: {
172  Building* building;
173  ::deserialize(ar, &building);
174  if(building == nullptr)
175  throw ServerException("Unknown building");
176 
177  UnitType* unit_type;
178  ::deserialize(ar, &unit_type);
179  if(unit_type == nullptr)
180  throw ServerException("Unknown unit type");
181 
182  // Must control building
183  if(building->get_owner() != selected_nation)
184  throw ServerException("Nation does not control building");
185 
186  // TODO: Check nation can build this unit
187 
188  // Tell the building to build this specific unit type
189  building->working_unit_type = unit_type;
190  building->req_goods_for_unit = unit_type->req_goods;
191  print_info("New order for building; build unit [%s]", unit_type->ref_name.c_str());
192  } break;
193  // Client tells server to build new outpost, the location (& type) is provided by
194  // the client and the rest of the fields are filled by the server
195  case ActionType::BUILDING_ADD: {
196  Building* building = new Building();
197  ::deserialize(ar, building);
198 
199  g_world->insert(building);
200  print_info("[%s] has built a [%s]", selected_nation->ref_name.c_str(), building->type->ref_name.c_str());
201 
202  // Rebroadcast
203  broadcast(Action::BuildingAdd::form_packet(building));
204  } break;
205  // Client tells server that it wants to colonize a province, this can be rejected
206  // or accepted, client should check via the next PROVINCE_UPDATE action
207  case ActionType::PROVINCE_COLONIZE: {
208  Province* province;
209  ::deserialize(ar, &province);
210 
211  if(province == nullptr)
212  throw ServerException("Unknown province");
213 
214  // Must not be already owned
215  if(province->owner != nullptr)
216  throw ServerException("Province already has an owner");
217 
218  province->owner = selected_nation;
219 
220  // Rebroadcast
221  broadcast(packet);
222  } break;
223  // Simple IRC-like chat messaging system
224  case ActionType::CHAT_MESSAGE: {
225  std::string msg;
226  ::deserialize(ar, &msg);
227  print_info("Message: %s\n", msg.c_str());
228 
229  // Rebroadcast
230  broadcast(packet);
231  } break;
232  // Client changes it's approval on certain treaty
233  case ActionType::CHANGE_TREATY_APPROVAL: {
234  Treaty* treaty;
235  ::deserialize(ar, &treaty);
236  if(treaty == nullptr)
237  throw ServerException("Treaty not found");
238 
239  TreatyApproval approval;
240  ::deserialize(ar, &approval);
241 
242  print_info("[%s] approves treaty [%s]? %s!", selected_nation->ref_name.c_str(), treaty->name.c_str(), (approval == TreatyApproval::ACCEPTED) ? "YES" : "NO");
243 
244  if(!treaty->does_participate(*selected_nation))
245  throw ServerException("Nation does not participate in treaty");
246 
247  // Rebroadcast
248  broadcast(packet);
249  } break;
250  // Client sends a treaty to someone
251  case ActionType::DRAFT_TREATY: {
252  Treaty* treaty = new Treaty();
253  ::deserialize(ar, &treaty->clauses);
254  ::deserialize(ar, &treaty->name);
255  ::deserialize(ar, &treaty->sender);
256 
257  // Validate data
258  if(!treaty->clauses.size())
259  throw ServerException("Clause-less treaty");
260  if(treaty->sender == nullptr)
261  throw ServerException("Treaty has invalid ends");
262 
263  // Obtain participants of the treaty
264  std::set<Nation*> approver_nations = std::set<Nation*>();
265  for(auto& clause : treaty->clauses) {
266  if(clause->receiver == nullptr || clause->sender == nullptr)
267  throw ServerException("Invalid clause receiver/sender");
268 
269  approver_nations.insert(clause->receiver);
270  approver_nations.insert(clause->sender);
271  }
272 
273  print_info("Participants of treaty [%s]", treaty->name.c_str());
274  // Then fill as undecided (and ask nations to sign this treaty)
275  for(auto& nation : approver_nations) {
276  treaty->approval_status.push_back(std::make_pair(nation, TreatyApproval::UNDECIDED));
277  print_info("- [%s]", nation->ref_name.c_str());
278  }
279 
280  // The sender automatically accepts the treaty (they are the ones who drafted it)
281  for(auto& status : treaty->approval_status) {
282  if(status.first == selected_nation) {
283  status.second = TreatyApproval::ACCEPTED;
284  break;
285  }
286  }
287 
288  g_world->insert(treaty);
289 
290  // Rebroadcast to client
291  // We are going to add a treaty to the client
292  Archive tmp_ar = Archive();
293  action = ActionType::TREATY_ADD;
294  ::serialize(tmp_ar, &action);
295  ::serialize(tmp_ar, treaty);
296  packet.data(tmp_ar.get_buffer(), tmp_ar.size());
297  broadcast(packet);
298  } break;
299  // Client takes a descision
300  case ActionType::NATION_TAKE_DESCISION: {
301  // Find event by reference name
302  std::string event_ref_name;
303  ::deserialize(ar, &event_ref_name);
304  auto event = std::find_if(g_world->events.begin(), g_world->events.end(), [&event_ref_name](const Event* e) {
305  return e->ref_name == event_ref_name;
306  });
307  if(event == g_world->events.end())
308  throw ServerException("Event not found");
309 
310  // Find descision by reference name
311  std::string descision_ref_name;
312  ::deserialize(ar, &descision_ref_name);
313  auto descision = std::find_if((*event)->descisions.begin(), (*event)->descisions.end(), [&descision_ref_name](const Descision& d) {
314  return d.ref_name == descision_ref_name;
315  });
316  if(descision == (*event)->descisions.end()) {
317  throw ServerException("Descision " + descision_ref_name + " not found");
318  }
319 
320  (*event)->take_descision(*selected_nation, *descision);
321  print_info("Event [%s] + descision [%s] taken by [%s]",
322  event_ref_name.c_str(),
323  descision_ref_name.c_str(),
324  selected_nation->ref_name.c_str()
325  );
326  } break;
327  // The client selects a nation
328  case ActionType::SELECT_NATION: {
329  Nation* nation;
330  ::deserialize(ar, &nation);
331  if(nation == nullptr)
332  throw ServerException("Unknown nation");
333  nation->ai_do_policies = false;
334  nation->ai_do_research = false;
335  nation->ai_do_diplomacy = false;
336  nation->ai_do_cmd_troops = false;
337  nation->ai_do_unit_production = false;
338  nation->ai_do_build_production = false;
339  nation->ai_handle_treaties = false;
340  nation->ai_handle_events = false;
341  selected_nation = nation;
342  print_info("Nation [%s] selected by client %zu", selected_nation->ref_name.c_str(), (size_t)id);
343  } break;
344  case ActionType::DIPLO_INC_RELATIONS: {
345  Nation* target;
346  ::deserialize(ar, &target);
347  selected_nation->increase_relation(*target);
348  } break;
349  case ActionType::DIPLO_DEC_RELATIONS: {
350  Nation* target;
351  ::deserialize(ar, &target);
352  selected_nation->decrease_relation(*target);
353  } break;
354  case ActionType::DIPLO_DECLARE_WAR: {
355  Nation* target;
356  ::deserialize(ar, &target);
357  selected_nation->declare_war(*target);
358  } break;
359  case ActionType::FOCUS_TECH: {
360  Technology* technology;
361  ::deserialize(ar, &technology);
362  if(technology == nullptr)
363  throw ServerException("Unknown technology");
364 
365  if(!selected_nation->can_research(technology))
366  throw ServerException("Can't research tech at the moment");
367 
368  selected_nation->focus_tech = technology;
369  } break;
370  case ActionType::AI_CONTROL: {
371  ::deserialize(ar, &selected_nation->ai_do_build_production);
372  ::deserialize(ar, &selected_nation->ai_do_cmd_troops);
373  ::deserialize(ar, &selected_nation->ai_do_diplomacy);
374  ::deserialize(ar, &selected_nation->ai_do_policies);
375  ::deserialize(ar, &selected_nation->ai_do_research);
376  ::deserialize(ar, &selected_nation->ai_do_unit_production);
377  ::deserialize(ar, &selected_nation->ai_handle_events);
378  ::deserialize(ar, &selected_nation->ai_handle_treaties);
379  } break;
380  // Nation and province addition and removals are not allowed to be done by clients
381  default:
382  break;
383  }
384  }
385 
386  ar.buffer.clear();
387  ar.rewind();
388 
389  // After reading everything we will send our queue appropriately to the client
390  std::scoped_lock lock(cl.packets_mutex);
391  for(auto& packet : cl.packets) {
392  packet.stream = UnifiedRender::Networking::SocketStream(conn_fd);
393  packet.send();
394  }
395  cl.packets.clear();
396  }
397  } catch(ServerException& e) {
398  print_error("ServerException: %s", e.what());
399  } catch(UnifiedRender::Networking::SocketException& e) {
400  print_error("SocketException: %s", e.what());
401  } catch(SerializerException& e) {
402  print_error("SerializerException: %s", e.what());
403  }
404 
405  player_count--;
406 
407 #ifdef windows
408  print_error("WSA Code: %u", WSAGetLastError());
409  WSACleanup();
410 #endif
411 
412  // Unlock mutexes so we don't end up with weird situations... like deadlocks
413  cl.is_connected = false;
414 
415  std::scoped_lock lock(cl.packets_mutex, cl.pending_packets_mutex);
416  cl.packets.clear();
417  cl.pending_packets.clear();
418 
419  // Tell the remaining clients about the disconnection
420  {
421  UnifiedRender::Networking::Packet packet;
422  Archive ar = Archive();
423  ActionType action = ActionType::DISCONNECT;
424  ::serialize(ar, &action);
425  packet.data(ar.get_buffer(), ar.size());
426  broadcast(packet);
427  }
428 
429  print_info("Client disconnected");
430 #ifdef windows
431  shutdown(conn_fd, SD_BOTH);
432 #elif defined unix
433  shutdown(conn_fd, SHUT_RDWR);
434 #endif
435  }
436 }
