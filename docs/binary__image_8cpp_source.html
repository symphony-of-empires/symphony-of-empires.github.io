58 void BinaryImage::from_file(const UnifiedRender::IO::Path& path) {
59  int i_width, i_height, i_channels;
60 
61  // stbi can do the conversion to RGBA for us ;)
62  stbi_uc* c_buffer = stbi_load(Path::get(path.str.c_str()).c_str(), &i_width, &i_height, &i_channels, 4);
63  if(c_buffer == nullptr) {
64  throw BinaryImageException(path.str, std::string() + "Image load error: " + stbi_failure_reason());
65  }
66  buffer.reset((uint32_t*)c_buffer);
67 
68  width = (size_t)i_width;
69  height = (size_t)i_height;
70 }
