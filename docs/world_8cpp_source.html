66 const std::vector<const Tile*> Tile::get_neighbours(const World& world) const {
67  std::vector<const Tile*> tiles;
68  const size_t idx = world.get_id(this);
69 
70  // Up
71  if(idx > world.width) {
72  tiles.push_back(&world.tiles[idx - world.width]);
73  }
74  // Down
75  if(idx < (world.width * world.height) - world.width) {
76  tiles.push_back(&world.tiles[idx + world.width]);
77  }
78  // Left
79  if(idx > 1) {
80  tiles.push_back(&world.tiles[idx - 1]);
81  }
82  // Right
83  if(idx < (world.width * world.height) - 1) {
84  tiles.push_back(&world.tiles[idx + 1]);
85  }
86  return tiles;
87 }

422 void World::load_initial(void) {
423  const std::vector<std::string> init_files = {
424  "terrain_types",
425  "ideologies", "cultures", "nations", "unit_traits", "building_types",
426  "technology", "religions", "pop_types", "good_types", "industry_types",
427  "unit_types", "boat_types", "provinces", "init"
428  };
429  lua_exec_all_of(*this, init_files, "lua/entities");
430 
431  // Shrink normally-not-resized vectors to give back memory to the OS
432  UnifiedRender::Log::debug("game", UnifiedRender::Locale::translate("Shrink normally-not-resized vectors to give back memory to the OS"));
433  provinces.shrink_to_fit();
434  nations.shrink_to_fit();
435  goods.shrink_to_fit();
436  unit_types.shrink_to_fit();
437  cultures.shrink_to_fit();
438  religions.shrink_to_fit();
439  pop_types.shrink_to_fit();
440  ideologies.shrink_to_fit();
441  building_types.shrink_to_fit();
442 
443  // Build a lookup table for super fast speed on finding provinces
444  // 16777216 * 4 = c.a 64 MB, that quite a lot but we delete the table after anyways
445  UnifiedRender::Log::debug("game", UnifiedRender::Locale::translate("Building the province lookup table"));
446  std::vector<Province::Id> province_color_table(16777216, (Province::Id)-1);
447  for(auto& province : provinces) {
448  province_color_table[province->color & 0xffffff] = get_id(province);
449  }
450 
451  // Associate tiles with province
452  std::unique_ptr<BinaryImage> div = std::unique_ptr<BinaryImage>(new BinaryImage(Path::get("map/provinces.png")));
453  width = div->width;
454  height = div->height;
455 
456  // Sea is <= sea_level
457  // Rivers == sea_level + 1
458  // Land is > sea_level + 1
459  const size_t total_size = width * height;
460  sea_level = 126;
461  tiles = new Tile[total_size];
462  if(tiles == nullptr) {
463  throw std::runtime_error("Out of memory");
464  }
465 
466  if(div->width != width || div->height != height) {
467  throw std::runtime_error("Province map size mismatch");
468  }
469 
470  if(1) {
471  for(auto& province : provinces) {
472  province->n_tiles = 0;
473  }
474 
475  for(unsigned int i = 0; i < total_size; ) {
476  const Province::Id province_id = province_color_table[div->buffer.get()[i] & 0xffffff];
477  if(province_id >= provinces.size()) {
478  i++;
479  continue;
480  }
481 
482  while(div->buffer.get()[i] == provinces[province_id]->color) {
483  provinces[province_id]->n_tiles++;
484  i++;
485  }
486  }
487 
488  for(unsigned int i = 0; i < provinces.size(); i++) {
489  auto& province = *provinces[i];
490  if(!province.n_tiles) {
491  remove(&province);
492 
493  for(auto& nation : nations) {
494  nation->owned_provinces.erase(&province);
495  if(nation->capital == &province) {
496  nation->capital = nullptr;
497  }
498  }
499 
500  for(size_t j = 0; j < buildings.size(); j++) {
501  auto& building = *buildings[j];
502 
503  if(building.province != &province) {
504  continue;
505  }
506 
507  remove(&building);
508  delete &building;
509  j--;
510  }
511 
512  delete &province;
513  i--;
514  continue;
515  }
516  }
517 
518  std::fill(province_color_table.begin(), province_color_table.end(), (Province::Id)-1);
519  for(auto& province : provinces) {
520  province_color_table[province->color & 0xffffff] = get_id(province);
521  }
522  }
523 
524  uint32_t checksum = 1;
525  for(auto& province : provinces) {
526  checksum += province->ref_name.at(0);
527  }
528  checksum *= width * height * provinces.size();
529 
530  bool recalc_province = true;
531  try {
532  Archive ar = Archive();
533  ar.from_file("province_cache.dat");
534 
535  // If the cache has the same checksum as us then we just have to read from the file
536  uint32_t cache_checksum;
537  ::deserialize(ar, &cache_checksum);
538  if(checksum == cache_checksum) {
539  recalc_province = false;
540  for(auto& province : provinces) {
