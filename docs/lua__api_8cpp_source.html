64 template<typename T>
65 const T* find_or_throw(const std::string& ref_name) {
66  const T* obj_to_find = nullptr;
67 
68  const auto& list = World::get_instance().get_list(obj_to_find);
69  const auto result = std::find_if(list.begin(), list.end(), [&ref_name](const auto& o) {
70  return (o->ref_name == ref_name);
71  });
72 
73  if(result == list.end()) {
74  throw LuaAPI::Exception(ref_name + " not found");
75  }
76  return (*result);
77 }

1213  if(!nation->exists()) {

1218  lua_pushstring(L, nation->ref_name.c_str());
1219  lua_pcall(L, 1, 1, 0);

1221  lua_pop(L, 1);

1225  has_fired = true;

1228

1230  lua_getglobal(L, event->do_event_function.c_str());
1231  lua_pushstring(L, nation->ref_name.c_str());
1232  lua_pcall(L, 1, 1, 0);
1233  is_multi = lua_tointeger(L, -1);
1234  lua_pop(L, 1);
1235

1241  local_event->ref_name += 'a' + (rand() % 26);
1242  }

1258  *event = orig_event;
1259  }
