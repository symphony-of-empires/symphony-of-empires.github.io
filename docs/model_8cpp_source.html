125 const UnifiedRender::Model& UnifiedRender::ModelManager::load_wavefront(const std::string& path) {
126  class WavefrontFace {
127  public:
128  // Indexes to the actual points
129  std::vector<int> vertices, texcoords, normals;
130  };
131 
132  class WavefrontObj {
133  public:
134  WavefrontObj(const std::string& _name) : name(_name) {};
135  std::string name;
136  std::vector<WavefrontFace> faces;
137  const UnifiedRender::Material* material = nullptr;
138  };
139 
140  std::ifstream file(path);
141  std::string line;
142 
143  // Recollect all data from the .OBJ file
144  std::vector<glm::vec3> vertices, normals;
145  std::vector<glm::vec2> texcoords;
146 
147  std::vector<WavefrontObj> objects;
148  objects.push_back(WavefrontObj("default"));
149  while(std::getline(file, line)) {
150  // Skip whitespace
151  size_t len = line.find_first_not_of(" \t");
152  if(len != std::string::npos) {
153  line = line.substr(len, line.length() - len);
154  }
155 
156  // Comment
157  if(line[0] == '#' || line.empty()) {
158  continue;
159  }
160 
161  std::istringstream sline(line);
162  std::string cmd;
163  sline >> cmd;
164  if(cmd == "mtllib") {
165  std::string name;
166  sline >> name;
167  State::get_instance().material_man->load_wavefront(Path::get("3d/" + name));
168  } else if(cmd == "usemtl") {
169  std::string name;
170  sline >> name;
171  objects.front().material = &UnifiedRender::State::get_instance().material_man->load(name);
172  } else if(cmd == "o") {
173  std::string name;
174  sline >> name;
175  objects.push_back(WavefrontObj(name));
176  } else if(cmd == "v") {
177  glm::vec3 vert;
178  sline >> vert.x >> vert.y >> vert.z;
179  vertices.push_back(vert);
180  } else if(cmd == "vt") {
181  glm::vec2 tex;
182  sline >> tex.x >> tex.y;
183  texcoords.push_back(tex);
184  } else if(cmd == "vn") {
185  glm::vec3 norm;
186  sline >> norm.x >> norm.y >> norm.z;
187  normals.push_back(norm);
188  } else if(cmd == "f") {
189  WavefrontFace face = WavefrontFace();
190  while(sline.peek() != -1) {
191  // Assemble faces - allowing for any number of vertices
192  // (and respecting the optional-ity of vt and vn fields)
193  int value;
194  char ch;
195 
196  // Vertices
197  sline >> value;
198  face.vertices.push_back(value);
199 
200  ch = sline.peek();
201  if(ch == '/') {
202  sline >> ch;
203 
204  // Texcoords
205  sline >> value;
206  face.texcoords.push_back(value);
207 
208  ch = sline.peek();
209  if(ch == '/') {
210  sline >> ch;
211 
212  // Normals
213  sline >> value;
214  face.normals.push_back(value);
215 
216  ch = sline.peek();
217  if(ch == '/') {
218  print_error("Invalid face declaration");
219  break;
220  }
221  }
222  }
223  }
224 
225  if(face.vertices.size() < 3) {
226  print_error("Cannot create polygon - malformed face?");
227  } else {
228  objects.front().faces.push_back(face);
229  }
230  } else if(cmd == "s") {
231  std::string light_mode;
232  sline >> light_mode;
233 
234  if(light_mode != "off") {
235  print_error("Unsupported light mode %s", light_mode.c_str());
236  }
237  } else {
238  print_error("Unsupported command %s", cmd.c_str());
239  }
240  }
241 
242  // Convert objects into (UnifiedRender) simple objects so we can now use them
243  UnifiedRender::Model* final_model = new UnifiedRender::Model();
244  for(std::vector<WavefrontObj>::const_iterator obj = objects.cbegin(); obj != objects.cend(); obj++) {
245  // Fill up the trigonometric buffers, we will first read all the faces and make them separate
246  std::vector<std::vector<UnifiedRender::MeshData<glm::vec3, glm::vec2>>> clusters{};
247  for(std::vector<WavefrontFace>::const_iterator face = (*obj).faces.cbegin(); face != (*obj).faces.cend(); face++) {
248  std::vector<UnifiedRender::MeshData<glm::vec3, glm::vec2>> cluster{};
249 
250  // The faces dictate indices for the vertices and we will also subtract 1 because the indexing is 0 based
251  if((*face).vertices.size() == (*face).texcoords.size()) {
252  for(unsigned int i = 0; i < (*face).vertices.size(); i++) {
253  cluster.push_back(UnifiedRender::MeshData<glm::vec3, glm::vec2>(
254  glm::vec3(vertices[(*face).vertices[i] - 1]),
255  glm::vec2(texcoords[(*face).texcoords[i] - 1])
256  ));
257  }
258  } else {
259  for(unsigned int i = 0; i < (*face).vertices.size(); i++) {
260  cluster.push_back(UnifiedRender::MeshData<glm::vec3, glm::vec2>(
261  glm::vec3(vertices[(*face).vertices[i] - 1]),
262  glm::vec2(0.f, 0.f)
263  ));
264  }
265  }
266  clusters.push_back(cluster);
267  }
268 
269  // TODO: This can be optimized even more
270 
271  // Now it's time to merge it all into a single mesh
272 
273  // It's time to merge clusters which share triangle nodes, in such case we reorganize the triangles so they
274  // are to "follow" the triangle fan
275  std::vector<std::vector<UnifiedRender::MeshData<glm::vec3, glm::vec2>>>::const_iterator cluster;
276  for(cluster = clusters.cbegin(); cluster != clusters.cend(); cluster++) {
277  UnifiedRender::SimpleModel* model = new UnifiedRender::SimpleModel(UnifiedRender::MeshMode::TRIANGLE_FAN);
278 
279  std::vector<UnifiedRender::MeshData<glm::vec3, glm::vec2>>::const_iterator v1;
280  for(v1 = (*cluster).cbegin(); v1 != (*cluster).cend(); v1++) {
281  model->buffer.push_back(*v1);
282  }
283 
284  print_info("Created new SimpleModel with %zu vertices", model->buffer.size());
285  model->material = (*obj).material;
286  model->upload();
287  final_model->simple_models.push_back(model);
288  }
289  }
290 
291  models[path] = final_model;
292  return *final_model;
293 }
